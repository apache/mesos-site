24 // Provides an abstraction that can take a standard function object
25 // and defer it without needing a process. Each converted function
26 // object will get execute serially with respect to one another when
27 // invoked.
28 class Executor

30 public:

34  }

40  }

42  void stop()

46  // TODO(benh): Note that this doesn't wait because that could
47  // cause a deadlock ... thus, the semantics here are that no more
48  // dispatches will occur after this function returns but one may
49  // be occurring concurrently.
50  }
