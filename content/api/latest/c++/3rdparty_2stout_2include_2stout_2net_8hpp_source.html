69 // Initializes libraries that net:: functions depend on, in a
70 // thread-safe way. This does not have to be called explicitly by
71 // the user of any functions in question. They will call this
72 // themselves by need.
73 inline void initialize()
74 {
75  // We use a static struct variable to initialize in a thread-safe
76  // way, at least with respect to calls within net::*, since there
77  // is no way to guarantee that another library is not concurrently
78  // initializing CURL. Thread safety is provided by the fact that
79  // the value 'curl' should get constructed (i.e., the CURL
80  // constructor invoked) in a thread safe way (as of GCC 4.3 and
81  // required for C++11).
82  struct CURL
83  {
84  CURL()
85  {
86  // This is the only one function in libcurl that is not deemed
87  // thread-safe. (And it must be called at least once before any
88  // other libcurl function is used.)
89  curl_global_init(CURL_GLOBAL_ALL);
90  }
91  };
92 
93  static CURL curl;
94 }

97 // Downloads the header of the specified HTTP URL with a HEAD request
98 // and queries its "content-length" field. (Note that according to the
99 // HTTP specification there is no guarantee that this field contains
100 // any useful value.)
101 inline Try<Bytes> contentLength(const std::string& url)
102 {
103  initialize();
104 
105  CURL* curl = curl_easy_init();
106  if (curl == nullptr) {
107  curl_easy_cleanup(curl);
108  return Error("Failed to initialize libcurl");
109  }
110 
111  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
112  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
113  curl_easy_setopt(curl, CURLOPT_HEADER, 1);
114  curl_easy_setopt(curl, CURLOPT_NOBODY, 1);
115 
116  CURLcode curlErrorCode = curl_easy_perform(curl);
117  if (curlErrorCode != 0) {
118  curl_easy_cleanup(curl);
119  return Error(curl_easy_strerror(curlErrorCode));
120  }
121 
122  double result;
123  curl_easy_getinfo(curl, CURLINFO_CONTENT_LENGTH_DOWNLOAD, &result);
124 
125  curl_easy_cleanup(curl);
126 
127  if (result < 0) {
128  return Error("No URL content-length available");
129  }
130 
131  return Bytes(uint64_t(result));
132 }

135 // Returns the HTTP response code resulting from attempting to
136 // download the specified HTTP or FTP URL into a file at the specified
137 // path.
138 inline Try<int> download(const std::string& url, const std::string& path)
139 {
140  initialize();
141 
142  Try<int_fd> fd = os::open(
143  path,
144  O_CREAT | O_WRONLY | O_CLOEXEC,
145  S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
146 
147  if (fd.isError()) {
148  return Error(fd.error());
149  }
150 
151  CURL* curl = curl_easy_init();
152 
153  if (curl == nullptr) {
154  curl_easy_cleanup(curl);
155  os::close(fd.get());
156  return Error("Failed to initialize libcurl");
157  }
158 
159  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
160  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, nullptr);
161  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
162 
163  // We don't bother introducing a `os::fdopen` since this is the only place
164  // we use `fdopen` in the entire codebase as of writing this comment.
165 #ifdef __WINDOWS__
166  FILE* file = ::_fdopen(fd->crt(), "w");
167 #else
168  FILE* file = ::fdopen(fd.get(), "w");
169 #endif
170  if (file == nullptr) {
171  curl_easy_cleanup(curl);
172  os::close(fd.get());
173  return ErrnoError("Failed to open file handle of '" + path + "'");
174  }
175  curl_easy_setopt(curl, CURLOPT_WRITEDATA, file);
176 
177  CURLcode curlErrorCode = curl_easy_perform(curl);
178  if (curlErrorCode != 0) {
179  curl_easy_cleanup(curl);
180  fclose(file);
181  return Error(curl_easy_strerror(curlErrorCode));
182  }
183 
184  long code;
185  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
186  curl_easy_cleanup(curl);
187 
188  if (fclose(file) != 0) {
189  return ErrnoError("Failed to close file handle of '" + path + "'");
190  }
191 
192  return Try<int>::some(code);
193 }

208 inline Try<std::string> hostname()
209 {

211 
212  if (gethostname(host, sizeof(host)) < 0) {
213  return ErrnoError();
214  }

233 // Returns a Try of the hostname for the provided IP. If the hostname
234 // cannot be resolved, then a string version of the IP address is
235 // returned.
236 //
237 // TODO(benh): Merge with `net::hostname`.
238 inline Try<std::string> getHostname(const IP& ip)
239 {
240  struct sockaddr_storage storage;
241  memset(&storage, 0, sizeof(storage));
242 
243  switch (ip.family()) {
244  case AF_INET: {
245  struct sockaddr_in addr;
246  memset(&addr, 0, sizeof(addr));
247  addr.sin_family = AF_INET;
248  addr.sin_addr = ip.in().get();
249  addr.sin_port = 0;
250 
251  memcpy(&storage, &addr, sizeof(addr));
252  break;
253  }
254  default: {
255  ABORT("Unsupported family type: " + stringify(ip.family()));
256  }
257  }
258 
259  char hostname[MAXHOSTNAMELEN];
260 
261  int error = getnameinfo(
262  (struct sockaddr*) &storage,
263 #ifdef __FreeBSD__
264  sizeof(struct sockaddr_in),
265 #else
266  sizeof(storage),
267 #endif
268  hostname,
269  MAXHOSTNAMELEN,
270  nullptr,
271  0,
272  0);
273 
274  if (error != 0) {
275  return Error(std::string(gai_strerror(error)));
276  }
277 
278  return std::string(hostname);
279 }

282 // Returns the names of all the link devices in the system.
283 inline Try<std::set<std::string>> links()
284 {

286  return Error("Not implemented");

295  if (ifa->ifa_name != nullptr) {

298  }

306 // Returns a Try of the IP for the provided hostname or an error if no IP is
307 // obtained.
308 inline Try<IP> getIP(const std::string& hostname, int family)
309 {
310  struct addrinfo hints = createAddrInfo(SOCK_STREAM, family, 0);
311  struct addrinfo* result = nullptr;
312 
313  int error = getaddrinfo(hostname.c_str(), nullptr, &hints, &result);
314 
315  if (error != 0) {
316  return Error(gai_strerror(error));
317  }
318 
319  if (result->ai_addr == nullptr) {
320  freeaddrinfo(result);
321  return Error("No addresses found");
322  }
323 
324  Try<IP> ip = IP::create(*result->ai_addr);
325 
326  if (ip.isError()) {
327  freeaddrinfo(result);
328  return Error("Unsupported family type");
329  }
330 
331  freeaddrinfo(result);
332  return ip.get();
333 }
