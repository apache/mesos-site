52 // Helper that captures information about a stack to be used when
53 // invoking clone.
54 class Stack
55 {
56 public:
57  // 8 MiB is the default for "ulimit -s" on OSX and Linux.
58  static constexpr size_t DEFAULT_SIZE = 8 * 1024 * 1024;
59 
60  // Allocate a stack.
61  static Try<Stack> create(size_t size)
62  {
63  Stack stack(size);
64 
65  // Allocate and align the memory to 16 bytes.
66  // x86, x64, and AArch64/ARM64 all expect a 16 byte aligned stack.
67  // ARM64/aarch64 enforces the alignment where x86/x64 does not.
68  // Without this alignment Mesos will crash with a SIGBUS on ARM64/aarch64.
69  int err = ::posix_memalign(
70  reinterpret_cast<void**>(&stack.address),
71  os::pagesize(),
72  stack.size);
73  if (err) {
74  return ErrnoError("Failed to allocate and align stack");
75  }
76 
77  return stack;
78  }
79 
80  // Explicitly free the stack.
81  // The destructor won't free the allocated stack.
82  void deallocate()
83  {
84  ::free(address);
85  address = nullptr;
86  size = 0;
87  }
88 
89  // Stack grows down, return the first usable address.
90  char* start()
91  {
92  return address + size;
93  }
94 
95 private:
96  explicit Stack(size_t size_) : size(size_) {}
97 
98  size_t size;
99  char* address;
100 };

103 inline pid_t clone(
104  const lambda::function<int()>& func,
105  int flags,
106  Option<Stack> stack = None())
107 {
108  // Stack for the child.
109  //
110  // NOTE: We need to allocate the stack dynamically. This is because
111  // glibc's 'clone' will modify the stack passed to it, therefore the
112  // stack must NOT be shared as multiple 'clone's can be invoked
113  // simultaneously.
114 
115  bool cleanup = false;
116  if (stack.isNone()) {
117  Try<Stack> _stack = Stack::create(Stack::DEFAULT_SIZE);
118  if (_stack.isError()) {
119  return -1;
120  }
121  stack = _stack.get();
122  cleanup = true;
123  }
124 
125  pid_t pid = ::clone(childMain, stack->start(), flags, (void*) &func);
126 
127  // Given we allocated the stack ourselves, there are two
128  // circumstances where we need to delete the allocated stack to
129  // avoid a memory leak:
130  //
131  // (1) Failed to clone.
132  //
133  // (2) CLONE_VM is not set implying ::clone will create a process
134  // which runs in its own copy of the memory space of the
135  // calling process. If CLONE_VM is set ::clone will create a
136  // thread which runs in the same memory space with the calling
137  // process, in which case we don't want to call delete!
138  if (cleanup && (pid < 0 || !(flags & CLONE_VM))) {
139  stack->deallocate();
140  }
141 
142  return pid;
143 }

146 inline Result<Process> process(pid_t pid)
147 {
148  // Page size, used for memory accounting.
149  static const size_t pageSize = os::pagesize();
150 
151  // Number of clock ticks per second, used for cpu accounting.
152  static const long ticks = sysconf(_SC_CLK_TCK);
153  if (ticks <= 0) {
154  return Error("Failed to get sysconf(_SC_CLK_TCK)");
155  }
156 
157  const Result<proc::ProcessStatus> status = proc::status(pid);
158 
159  if (status.isError()) {
160  return Error(status.error());
161  }
162 
163  if (status.isNone()) {
164  return None();
165  }
166 
167  // There are known bugs with invalid utime / stime values coming
168  // from /proc/<pid>/stat on some Linux systems.
169  // See the following thread for details:
170  // http://mail-archives.apache.org/mod_mbox/incubator-mesos-dev/
171  // 201307.mbox/%3CCA+2n2er-Nemh0CsKLbHRkaHd=YCrNt17NLUPM2=TtEfsKOw4
172  // Rg@mail.gmail.com%3E
173  // These are similar reports:
174  // http://lkml.indiana.edu/hypermail/linux/kernel/1207.1/01388.html
175  // https://bugs.launchpad.net/ubuntu/+source/linux/+bug/1023214
176  Try<Duration> utime = Duration::create(status.get().utime / (double) ticks);
177  Try<Duration> stime = Duration::create(status.get().stime / (double) ticks);
178 
179  // The command line from 'status.get().comm' is only "arg0" from
180  // "argv" (i.e., the canonical executable name). To get the entire
181  // command line we grab '/proc/[pid]/cmdline'.
182  Result<std::string> cmdline = proc::cmdline(pid);
183 
184  return Process(status.get().pid,
185  status.get().ppid,
186  status.get().pgrp,
187  status.get().session,
188  Bytes(status.get().rss * pageSize),
189  utime.isSome() ? utime.get() : Option<Duration>::none(),
190  stime.isSome() ? stime.get() : Option<Duration>::none(),
191  cmdline.isSome() ? cmdline.get() : status.get().comm,
192  status.get().state == 'Z');
193 }

202 // Returns the total size of main and free memory.
203 inline Try<Memory> memory()
204 {
205  Memory memory;
206 
207  struct sysinfo info;
208  if (sysinfo(&info) != 0) {
209  return ErrnoError();
210  }
211 
212 # if LINUX_VERSION_CODE >= KERNEL_VERSION(2, 3, 23)
213  memory.total = Bytes(info.totalram * info.mem_unit);
214  memory.free = Bytes(info.freeram * info.mem_unit);
215  memory.totalSwap = Bytes(info.totalswap * info.mem_unit);
216  memory.freeSwap = Bytes(info.freeswap * info.mem_unit);
217 # else
218  memory.total = Bytes(info.totalram);
219  memory.free = Bytes(info.freeram);
220  memory.totalSwap = Bytes(info.totalswap);
221  memory.freeSwap = Bytes(info.freeswap);
222 # endif
223 
224  return memory;
225 }
