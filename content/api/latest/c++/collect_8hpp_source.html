100 
101  void waited(const Future<T>& future)
102  {
103  if (future.isFailed()) {
104  promise->fail("Collect failed: " + future.failure());
105  terminate(this);
106  } else if (future.isDiscarded()) {
107  promise->fail("Collect failed: future discarded");
108  terminate(this);
109  } else {
110  CHECK_READY(future);
111  ready += 1;
112  if (ready == futures.size()) {
113  std::list<T> values;
114  foreach (const Future<T>& future, futures) {
115  values.push_back(future.get());
116  }
117  promise->set(values);
118  terminate(this);
119  }
120  }
121  }
122 
123  const std::list<Future<T>> futures;
124  Promise<std::list<T>>* promise;
125  size_t ready;
126 };

188 template <typename T>
189 inline Future<std::list<T>> collect(
190  const std::list<Future<T>>& futures)
191 {
192  if (futures.empty()) {
193  return std::list<T>();
194  }
195 
196  Promise<std::list<T>>* promise = new Promise<std::list<T>>();
197  Future<std::list<T>> future = promise->future();
198  spawn(new internal::CollectProcess<T>(futures, promise), true);
199  return future;
200 }

203 template <typename... Ts>
204 Future<std::tuple<Ts...>> collect(const Future<Ts>&... futures)
205 {
206  std::list<Future<Nothing>> wrappers = {
207  futures.then([]() { return Nothing(); })...
208  };
209 
210  // TODO(klueska): Unfortunately, we have to use a lambda followed
211  // by a bind here because of a bug in gcc 4.8 to handle variadic
212  // parameters in lambdas:
213  // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=47226
214  auto f = [](const Future<Ts>&... futures) {
215  return std::make_tuple(futures.get()...);
216  };
217 
218  return collect(wrappers)
219  .then(std::bind(f, futures...));
220 }
