76 // Helper for repeated field devolving to 'T1' from 'T2'.
77 template <typename T1, typename T2>

79  google::protobuf::RepeatedPtrField<T2> t2s)
80 {
81  google::protobuf::RepeatedPtrField<T1> t1s;
82 
83  foreach (const T2& t2, t2s) {
84  t1s.Add()->CopyFrom(devolve(t2));
85  }
86 
87  return t1s;
