92 // Helper for repeated field evolving to 'T1' from 'T2'.
93 template <typename T1, typename T2>
94 google::protobuf::RepeatedPtrField<T1> evolve(
95  google::protobuf::RepeatedPtrField<T2> t2s)
96 {
97  google::protobuf::RepeatedPtrField<T1> t1s;
98 
99  foreach (const T2& t2, t2s) {
100  t1s.Add()->CopyFrom(evolve(t2));
101  }
102 
103  return t1s;
104 }

141 // Before the v1 API we had REST endpoints that returned JSON. The JSON was not
142 // specified in any formal way, i.e., there were no protobufs which captured the
143 // structure. As part of the v1 API we introduced the Call/Response protobufs
144 // (see v1/master.proto and v1/agent.proto). This evolve variant transforms a
145 // JSON object that would have been returned from a particular REST endpoint to
146 // a `Response` protobuf suitable for returning from the new v1 API endpoints.
