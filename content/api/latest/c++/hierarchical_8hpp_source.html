340  struct Slave
341  {
342  // Total amount of regular *and* oversubscribed resources.
343  Resources total;
344 
345  // Regular *and* oversubscribed resources that are allocated.
346  //
347  // NOTE: We maintain multiple copies of each shared resource allocated
348  // to a slave, where the number of copies represents the number of times
349  // this shared resource has been allocated to (and has not been recovered
350  // from) a specific framework.
351  //
352  // NOTE: We keep track of slave's allocated resources despite
353  // having that information in sorters. This is because the
354  // information in sorters is not accurate if some framework
355  // hasn't reregistered. See MESOS-2919 for details.
356  Resources allocated;
357 
358  // We track the total and allocated resources on the slave, the
359  // available resources are computed as follows:
360  //
361  // available = total - allocated
362  //
363  // Note that it's possible for the slave to be over-allocated!
364  // In this case, allocated > total.
365  Resources available() const
366  {
367  // In order to subtract from the total,
368  // we strip the allocation information.
369  Resources allocated_ = allocated;
370  allocated_.unallocate();
371 
372  return total - allocated_;
373  }

381  // Represents a scheduled unavailability due to maintenance for a specific
382  // slave, and the responses from frameworks as to whether they will be able
383  // to gracefully handle this unavailability.
384  //
385  // NOTE: We currently implement maintenance in the allocator to be able to
386  // leverage state and features such as the FrameworkSorter and OfferFilter.
387  struct Maintenance
388  {
389  Maintenance(const Unavailability& _unavailability)
390  : unavailability(_unavailability) {}
391 
392  // The start time and optional duration of the event.
393  Unavailability unavailability;
394 
395  // A mapping of frameworks to the inverse offer status associated with
396  // this unavailability.
397  //
398  // NOTE: We currently lose this information during a master fail over
399  // since it is not persisted or replicated. This is ok as the new master's
400  // allocator will send out new inverse offers and re-collect the
401  // information. This is similar to all the outstanding offers from an old
402  // master being invalidated, and new offers being sent out.
403  hashmap<FrameworkID, mesos::allocator::InverseOfferStatus> statuses;
404 
405  // Represents the "unit of accounting" for maintenance. When a

437  hashmap<std::string, double> weights;

439  // Configured quota for each role, if any. Setting quota for a role
440  // changes the order that the role's frameworks are offered
441  // resources. Quota comes before fair share, hence setting quota moves
442  // the role's frameworks towards the front of the allocation queue.
443  //
444  // NOTE: We currently associate quota with roles, but this may
445  // change in the future.
446  hashmap<std::string, Quota> quotas;

452  Option<std::set<std::string>> fairnessExcludeResourceNames;

454  // There are two stages of allocation. During the first stage resources
455  // are allocated only to frameworks in roles with quota set. During the
456  // second stage remaining resources that would not be required to satisfy
457  // un-allocated quota are then allocated to all frameworks.
458  //
459  // Each stage comprises two levels of sorting, hence "hierarchical".
460  // Level 1 sorts across roles:
461  // Currently, only the allocated portion of the reserved resources are
462  // accounted for fairness calculation.
463  //
464  // TODO(mpark): Reserved resources should be accounted for fairness
465  // calculation whether they are allocated or not, since they model a long or
466  // forever running task. That is, the effect of reserving resources is
467  // equivalent to launching a task in that the resources that make up the
468  // reservation are not available to other roles as non-revocable.
469  //
470  // Level 2 sorts across frameworks within a particular role:
471  // Reserved resources at this level are, and should be accounted for
472  // fairness calculation only if they are allocated. This is because
473  // reserved resources are fairly shared across the frameworks in the role.
474  //
475  // The allocator relies on `Sorter`s to employ a particular sorting
476  // algorithm. Each level has its own sorter and hence may have different
477  // fairness calculations.
478  //
479  // NOTE: The hierarchical allocator considers revocable resources as
480  // regular resources when doing fairness calculations.
481  //
482  // TODO(vinod): Consider using a different fairness algorithm for
483  // revocable resources.
484 
485  // A sorter for active roles. This sorter determines the order in which
486  // roles are allocated resources during Level 1 of the second stage.
487  process::Owned<Sorter> roleSorter;
488 
489  // A dedicated sorter for roles for which quota is set. This sorter
490  // determines the order in which quota'ed roles are allocated resources
491  // during Level 1 of the first stage. Quota'ed roles have resources
492  // allocated up to their alloted quota (the first stage) prior to
493  // non-quota'ed roles (the second stage).
494  //
495  // NOTE: A role appears in `quotaRoleSorter` if it has a quota (even if
496  // no frameworks are currently registered in that role). In contrast,
497  // `roleSorter` only contains entries for roles with one or more
498  // registered frameworks.
499  //
500  // NOTE: We do not include revocable resources in the quota role sorter,
501  // because the quota role sorter's job is to perform fair sharing between
502  // the quota roles as it pertains to their level of quota satisfaction.
503  // Since revocable resources do not increase a role's level of satisfaction
504  // toward its quota, we choose to exclude them from the quota role sorter.
505  process::Owned<Sorter> quotaRoleSorter;
506 
507  // A collection of sorters, one per active role. Each sorter determines
508  // the order in which frameworks that belong to the same role are allocated
509  // resources inside the role's share. These sorters are used during Level 2
510  // for both the first and the second stages.
511  hashmap<std::string, process::Owned<Sorter>> frameworkSorters;
512 
513  // Factory function for framework sorters.
514  const std::function<Sorter*()> frameworkSorterFactory;
