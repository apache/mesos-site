46 // Implementation of the JavaScript Object Notation (JSON) grammar
47 // using boost::variant. We explicitly define each "type" of the
48 // grammar, including 'true' (json::True), 'false' (json::False), and
49 // 'null' (json::Null), for clarity and also because boost::variant
50 // "picks" the wrong type when we try and use std::string, long (or
51 // int), double (or float), and bool all in the same variant (while it
52 // does work with explicit casts, it seemed bad style to force people
53 // to put those casts in place). We could have avoided using
54 // json::String or json::Number and just used std::string and double
55 // respectively, but we choose to include them for completeness
56 // (although, this does pay a 2x cost when compiling thanks to all the
57 // extra template instantiations).

275  // Returns true if and only if 'other' is contained by 'this'.
276  // 'Other' is contained by 'this' if the following conditions are
277  // fulfilled:
278  // 1. If 'other' is a JSON object, then 'this' is also a JSON
279  // object, all keys of 'other' are also present in 'this' and
280  // the value for each key in 'this' also contain the value for
281  // the same key in 'other', i.e. for all keys 'k' in 'other',
282  // 'this[k].contains(other[k])' is true.
283  // 2. If 'other' is a JSON array, 'this' is also a JSON array, the
284  // length of both arrays is the same and each element in 'this'
285  // also contains the element in 'other' at the same position,
286  // i.e. it holds that this.length() == other.length() and
287  // for each i, 0 <= i < this.length,
288  // 'this[i].contains(other[i])'.
289  // 3. For all other types, 'this' is of the same type as 'other' and
290  // 'this == other'.
291  // NOTE: For a given key 'k', if 'this[k] == null' then
292  // 'this.contains(other)' holds if either 'k' is not present in
293  // 'other.keys()' or 'other[k] == null'.
294  // Similarly, if 'other[k] == null', 'this.contains(other)' only if
295  // 'this[k] == null'. This is a consequence of the containment
296  // definition.
297  bool contains(const Value& other) const;
298 
299 private:
300  friend struct Comparator;

352 template <typename T>
353 Result<T> Object::find(const std::string& path) const
354 {
355  const std::vector<std::string> names = strings::split(path, ".", 2);
356 
357  if (names.empty()) {
358  return None();
359  }
360 
361  std::string name = names[0];
362 
363  // Determine if we have an array subscript. If so, save it but
364  // remove it from the name for doing the lookup.
365  Option<size_t> subscript = None();
366  size_t index = name.find('[');
367  if (index != std::string::npos) {
368  // Check for the closing bracket.
369  if (name.at(name.length() - 1) != ']') {
370  return Error("Malformed array subscript, expecting ']'");
371  }
372 
373  // Now remove the closing bracket (last character) and everything
374  // before and including the opening bracket.
375  std::string s = name.substr(index + 1, name.length() - index - 2);
376 
377  // Now numify the subscript.
378  Try<int> i = numify<int>(s);
379 
380  if (i.isError()) {
381  return Error("Failed to numify array subscript '" + s + "'");
382  } else if (i.get() < 0) {
383  return Error("Array subscript '" + s + "' must be >= 0");
384  }
385 
386  subscript = i.get();
387 
388  // And finally remove the array subscript from the name.
389  name = name.substr(0, index);
390  }
391 
392  std::map<std::string, Value>::const_iterator entry = values.find(name);
393 
394  if (entry == values.end()) {
395  return None();
396  }
397 
398  Value value = entry->second;
399 
400  if (subscript.isSome()) {
401  if (value.is<Array>()) {
402  Array array = value.as<Array>();
403  if (subscript.get() >= array.values.size()) {
404  return None();
405  }
406  value = array.values[subscript.get()];
407  } else if (value.is<Null>()) {
408  return None();
409  } else {
410  // TODO(benh): Use a visitor to print out the intermediate type.
411  return Error("Intermediate JSON value not an array");
412  }
413  }
414 
415  if (names.size() == 1) {
416  if (value.is<T>()) {
417  return value.as<T>();
418  } else if (value.is<Null>()) {
419  return None();
420  } else {
421  // TODO(benh): Use a visitor to print out the type found.
422  return Error("Found JSON value of wrong type");
423  }
424  }
425 
426  if (!value.is<Object>()) {
427  // TODO(benh): Use a visitor to print out the intermediate type.
428  return Error("Intermediate JSON value not an object");
429  }
430 
431  return value.as<Object>().find<T>(names[1]);
432 }

465 inline bool Value::ContainmentComparator::operator()(const Object& other) const
466 {
467  if (!self.is<Object>()) {
468  return false;
469  }
470 
471  // The empty set is contained in every set.
472  if (other.values.empty()) {
473  return true;
474  }
475 
476  const Object& _self = self.as<Object>();
477 
478  // All entries in 'other' should exists in 'self', which implies
479  // there should be at most as many entries in other as in self.
480  if (other.values.size() > _self.values.size()) {
481  return false;
482  }
483 
484  foreachpair (const std::string& key, const Value& value, other.values) {
485  auto _selfIterator = _self.values.find(key);
486 
487  if (_selfIterator == _self.values.end()) {
488  return false;
489  }
490 
491  if (!_selfIterator->second.contains(value)) {
492  return false;
493  }
494  }
495 
496  return true;
497 }

509 inline bool Value::ContainmentComparator::operator()(const Number& other) const
510 {
511  if (!self.is<Number>()) {
512  return false;
513  }
514 
515  // NOTE: For the following switch statements, we do not set a default
516  // case in order to leverage the compiler errors when the enumeration
517  // is augmented to find all the cases we need to provide.
518 
519  // NOTE: Using '==' is unsafe for unsigned-signed integer comparisons.
520  // The compiler will automatically cast the signed integer to an
521  // unsigned integer. i.e. If the signed integer was negative, it
522  // might be converted to a large positive number.
523  // Using the '==' operator *is* safe for double-integer comparisons.
524 
525  const Number& number = self.as<Number>();
526  switch (number.type) {
527  case Number::FLOATING:
528  switch (other.type) {
529  case Number::FLOATING:
530  return number.value == other.value;
531  case Number::SIGNED_INTEGER:
532  return number.value == other.signed_integer;
533  case Number::UNSIGNED_INTEGER:
534  return number.value == other.unsigned_integer;
535  }
536 
537  case Number::SIGNED_INTEGER:
538  switch (other.type) {
539  case Number::FLOATING:
540  return number.signed_integer == other.value;
541  case Number::SIGNED_INTEGER:
542  return number.signed_integer == other.signed_integer;
543  case Number::UNSIGNED_INTEGER:
544  // See note above for why this is not a simple '==' expression.
545  return number.signed_integer >= 0 &&
546  number.as<uint64_t>() == other.unsigned_integer;
547  }
548 
549  case Number::UNSIGNED_INTEGER:
550  switch (other.type) {
551  case Number::FLOATING:
552  return number.unsigned_integer == other.value;
553  case Number::SIGNED_INTEGER:
554  // See note above for why this is not a simple '==' expression.
555  return other.signed_integer >= 0 &&
556  number.unsigned_integer == other.as<uint64_t>();
557  case Number::UNSIGNED_INTEGER:
558  return number.unsigned_integer == other.unsigned_integer;
559  }
560  }
561 
562  UNREACHABLE();
563 }

737 inline Value convert(const picojson::value& value)
738 {
739  if (value.is<picojson::null>()) {
740  return Null();
741  } else if (value.is<bool>()) {
742  return Boolean(value.get<bool>());
743  } else if (value.is<picojson::value::object>()) {
744  Object object;
745  foreachpair (const std::string& name,
746  const picojson::value& v,
747  value.get<picojson::value::object>()) {
748  object.values[name] = convert(v);
749  }
750  return object;
751  } else if (value.is<picojson::value::array>()) {
752  Array array;
753  foreach (const picojson::value& v, value.get<picojson::value::array>()) {
754  array.values.push_back(convert(v));
755  }
756  return array;
757  } else if (value.is<int64_t>()) {
758  return Number(value.get<int64_t>());
759  } else if (value.is<double>()) {
760  return Number(value.get<double>());
761  } else if (value.is<std::string>()) {
762  return String(value.get<std::string>());
763  }
764  return Null();

770 inline Try<Value> parse(const std::string& s)
771 {
772  const char* parseBegin = s.c_str();
773  picojson::value value;
774  std::string error;
775 
776  // Because PicoJson supports repeated parsing of multiple objects/arrays in a
777  // stream, it will quietly ignore trailing non-whitespace characters. We would
778  // rather throw an error, however, so use `last_char` to check for this.
779  const char* lastVisibleChar =
780  parseBegin + s.find_last_not_of(strings::WHITESPACE);
781 
782  // Parse the string, returning a pointer to the character
783  // immediately following the last one parsed.
784  const char* parseEnd =
785  picojson::parse(value, parseBegin, parseBegin + s.size(), &error);
786 
787  if (!error.empty()) {
788  return Error(error);
789  } else if (parseEnd != lastVisibleChar + 1) {
790  return Error(
791  "Parsed JSON included non-whitespace trailing characters: "
792  + s.substr(parseEnd - parseBegin, lastVisibleChar + 1 - parseEnd));
793  }
794 
795  return internal::convert(value);
796 }
