150  void generate_keys_and_certs() {
151  // We store the allocated objects in these results so that we can
152  // have a consolidated 'cleanup()' function. This makes all the
153  // 'EXIT()' calls more readable and less error prone.
154  Result<EVP_PKEY*> private_key = None();
155  Result<X509*> certificate = None();
156  Result<EVP_PKEY*> scrap_key = None();
157  Result<X509*> scrap_certificate = None();
158 
159  auto cleanup = [&private_key, &certificate, &scrap_key, &scrap_certificate](
160  const Option<std::string> abort_message = None()) {
161  if (private_key.isSome()) { EVP_PKEY_free(private_key.get()); }
162  if (certificate.isSome()) { X509_free(certificate.get()); }
163  if (scrap_key.isSome()) { EVP_PKEY_free(scrap_key.get()); }
164  if (scrap_certificate.isSome()) { X509_free(scrap_certificate.get()); }
165 
166  // We abort here because failure during setup indicates that something
167  // is horribly and irrecoverably wrong.
168  if (abort_message.isSome()) {
169  ABORT(abort_message.get());
170  }
171  };
172 
173  // Generate the authority key.
174  private_key = process::network::openssl::generate_private_rsa_key();
175  if (private_key.isError()) {
176  cleanup("Could not generate private key: " + private_key.error());
177  }
178 
179  // Figure out the hostname that libprocess is advertising.
180  // Set the hostname of the certificate to this hostname so that
181  // hostname verification of the certificate will pass.
182  Try<std::string> hostname = net::getHostname(process::address().ip);
183  if (hostname.isError()) {
184  cleanup("Could not determine hostname of libprocess: " +
185  hostname.error());
186  }
187 
188  // Generate an authorized certificate.
189  certificate = process::network::openssl::generate_x509(
190  private_key.get(),
191  private_key.get(),
192  None(),
193  1,
194  365,
195  hostname.get(),
196  net::IP(process::address().ip));
197 
198  if (certificate.isError()) {
199  cleanup("Could not generate certificate: " + certificate.error());
200  }
201 
202  // Write the authority key to disk.
203  Try<Nothing> key_write =
204  process::network::openssl::write_key_file(private_key.get(), key_path());
205 
206  if (key_write.isError()) {
207  cleanup("Could not write private key to disk: " + key_write.error());
208  }
209 
210  // Write the authorized certificate to disk.
211  Try<Nothing> certificate_write =
212  process::network::openssl::write_certificate_file(
213  certificate.get(),
214  certificate_path());
215 
216  if (certificate_write.isError()) {
217  cleanup("Could not write certificate to disk: " +
218  certificate_write.error());
219  }
220 
221  // Generate a scrap key.
222  scrap_key = process::network::openssl::generate_private_rsa_key();
223  if (scrap_key.isError()) {
224  cleanup("Could not generate a scrap private key: " + scrap_key.error());
225  }
226 
227  // Write the scrap key to disk.
228  key_write = process::network::openssl::write_key_file(
229  scrap_key.get(),
230  scrap_key_path());
231 
232  if (key_write.isError()) {
233  cleanup("Could not write scrap key to disk: " + key_write.error());
234  }
235 
236  // Generate a scrap certificate.
237  scrap_certificate = process::network::openssl::generate_x509(
238  scrap_key.get(),
239  scrap_key.get());
240 
241  if (scrap_certificate.isError()) {
242  cleanup("Could not generate a scrap certificate: " +
243  scrap_certificate.error());
244  }
245 
246  // Write the scrap certificate to disk.
247  certificate_write = process::network::openssl::write_certificate_file(
248  scrap_certificate.get(),
249  scrap_certificate_path());
250 
251  if (certificate_write.isError()) {
252  cleanup("Could not write scrap certificate to disk: " +
253  certificate_write.error());
254  }
255 
256  // Since we successfully set up all our state, we call cleanup
257  // without an abort message (so as not to abort).
258  cleanup();
259  }

355  Result<std::string> path = os::realpath(BUILD_DIR);

362  // certificate verification can succeed.

364  full_environment["LIBPROCESS_IP"] = stringify(process::address().ip);
365 
366  return process::subprocess(
367  path::join(path.get(), "ssl-client"),
368  argv,
369  process::Subprocess::PIPE(),
370  process::Subprocess::PIPE(),
371  process::Subprocess::FD(STDERR_FILENO),
