62 // Represents a MAC address. A MAC address is a 48-bit unique
63 // identifier assigned to a network interface for communications on
64 // the physical network segment. We use a byte array (in transmission
65 // order) to represent a MAC address. For example, for a MAC address
66 // 01:23:34:67:89:ab, the format is shown as follows:
67 //
68 // MSB LSB
69 // | |
70 // v v
71 // +--------+--------+--------+--------+--------+--------+
72 // |bytes[0]|bytes[1]|bytes[2]|bytes[3]|bytes[4]|bytes[5]|
73 // +--------+--------+--------+--------+--------+--------+
74 //
75 // 01 : 23 : 45 : 67 : 89 : ab
76 // NOLINT(readability/ending_punctuation)
77 class MAC
78 {
79 public:
80  // Parse a MAC address (e.g., 01:23:34:67:89:ab).
81  static Try<MAC> parse(const std::string& s)
82  {
83  std::vector<std::string> tokens = strings::split(s, ":");
84  if (tokens.size() != 6) {
85  return Error("Invalid format. Expecting xx:xx:xx:xx:xx:xx");
86  }
87 
88  auto isValidHexDigit = [](char c) {
89  return (c >= '0' && c <= '9') ||
90  (c >= 'a' && c <= 'f') ||
91  (c >= 'A' && c <= 'F');
92  };
93 
94  uint8_t bytes[6];
95  for (size_t i = 0; i < 6; i++) {
96  if (tokens[i].size() != 2) {
97  return Error("Not a two digit hex number");
98  }
99 
100  if (!isValidHexDigit(tokens[i][0]) ||
101  !isValidHexDigit(tokens[i][1])) {
102  return Error("Not a valid hex number");
103  }
104 
105  const char* str = tokens[i].c_str();
106  char *endptr = nullptr;
107  unsigned long value = strtoul(str, &endptr, 16);
108 
109  assert(endptr == str + 2);
110  assert(value < 256);
111 
112  bytes[i] = static_cast<uint8_t>(value);
113  }
114 
115  return MAC(bytes);
116  }
117 
118  // Constructs a MAC address from a byte array.
119  explicit MAC(const uint8_t (&_bytes)[6])
120  {
121  for (size_t i = 0; i < 6; i++) {
122  bytes[i] = _bytes[i];
123  }
124  }

126  // Returns the byte at the given index. For example, for a MAC
127  // address 01:23:45:67:89:ab, mac[0] = 01, mac[1] = 23 and etc.
128  uint8_t operator[](size_t index) const

180 // Returns the MAC address of a given link device. The link device is
181 // specified using its name (e.g., eth0). Returns error if the link
182 // device is not found. Returns none if the link device is found, but
183 // does not have a MAC address (e.g., loopback).
184 inline Result<MAC> mac(const std::string& name)
185 {
186 #if !defined(__linux__) && !defined(__APPLE__) && !defined(__FreeBSD__)
187  return Error("Not implemented");
188 #else
189  struct ifaddrs* ifaddr = nullptr;
190  if (getifaddrs(&ifaddr) == -1) {
191  return ErrnoError();
192  }
193 
194  // Indicates whether the link device is found or not.
195  bool found = false;
196 
197  for (struct ifaddrs* ifa = ifaddr; ifa != nullptr; ifa = ifa->ifa_next) {
198  if (ifa->ifa_name != nullptr && !strcmp(ifa->ifa_name, name.c_str())) {
199  found = true;
200 
201 # if defined(__linux__)
202  if (ifa->ifa_addr != nullptr && ifa->ifa_addr->sa_family == AF_PACKET) {
203  struct sockaddr_ll* link = (struct sockaddr_ll*) ifa->ifa_addr;
204 
205  if (link->sll_halen == 6) {
206  struct ether_addr* addr = (struct ether_addr*) link->sll_addr;
207  MAC mac(addr->ether_addr_octet);
208 
209  // Ignore if the address is 0 so that the results are
210  // consistent on both OSX and Linux.
211  if (stringify(mac) == "00:00:00:00:00:00") {
212  continue;
213  }
214 
215  freeifaddrs(ifaddr);
216  return mac;
217  }
218  }
219 # elif defined(__APPLE__)
220  if (ifa->ifa_addr != nullptr && ifa->ifa_addr->sa_family == AF_LINK) {
221  struct sockaddr_dl* link = (struct sockaddr_dl*) ifa->ifa_addr;
222 
223  if (link->sdl_type == IFT_ETHER && link->sdl_alen == 6) {
224  struct ether_addr* addr = (struct ether_addr*) LLADDR(link);
225  MAC mac(addr->octet);
226 
227  freeifaddrs(ifaddr);
228  return mac;
229  }
230  }
231 # endif
232  }
233  }
234 
235  freeifaddrs(ifaddr);
236 
237  if (!found) {
238  return Error("Cannot find the link device");
239  }
240 
241  return None();
242 #endif
243 }
