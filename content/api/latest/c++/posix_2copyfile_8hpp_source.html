29 // This implementation works by running the `cp` command with some
30 // additional conditions to ensure we copy a single file only,
31 // from an absolute file path to another absolute file path.
32 //
33 // Directories are not supported as a destination path for two reasons:
34 // 1. No callers depended on that behavior,
35 // 2. Consistency with Windows implementation.
36 //
37 // Relative paths are not allowed, as these are resolved based on
38 // the current working directory and may be inconsistent.
39 inline Try<Nothing> copyfile(
40  const std::string& source, const std::string& destination)
41 {
42  // NOTE: We check the form of the path too in case it does not exist, and to
43  // prevent user error.
44  if (stat::isdir(source) || source.back() == '/') {
45  return Error("`source` was a directory");
46  }
47 
48  if (stat::isdir(destination) || destination.back() == '/') {
49  return Error("`destination` was a directory");
50  }
51 
52  if (!path::absolute(source)) {
53  return Error("`source` was a relative path");
54  }
55 
56  if (!path::absolute(destination)) {
57  return Error("`destination` was a relative path");
58  }
59 
60  const Option<int> status = os::spawn("cp", {"cp", source, destination});
61 
62  if (status.isNone()) {
63  return ErrnoError("os::spawn failed");
64  }
65 
66  if (!(WIFEXITED(status.get()) && WEXITSTATUS(status.get()) == 0)) {
67  return Error("cp failed with status: " + stringify(status.get()));
68  }
69 
70  return Nothing();
71 }
