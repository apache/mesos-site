110 // Suspends execution of the calling process until a child specified by `pid`
111 // has changed state. Unlike the POSIX standard function `::waitpid`, this
112 // function does not use -1 and 0 to signify errors and nonblocking return.
113 // Instead, we return `Result<pid_t>`:
114 // * In case of error, we return `Error` rather than -1. For example, we
115 // would return an `Error` in case of `EINVAL`.
116 // * In case of nonblocking return, we return `None` rather than 0. For
117 // example, if we pass `WNOHANG` in the `options`, we would expect 0 to be
118 // returned in the case that children specified by `pid` exist, but have
119 // not changed state yet. In this case we return `None` instead.
120 //
121 // NOTE: There are important differences between the POSIX and Windows
122 // implementations of this function:
123 // * On POSIX, `pid_t` is a signed number, but on Windows, PIDs are `DWORD`,
124 // which is `unsigned long`. Thus, if we use `DWORD` to represent the `pid`
125 // argument, we would not be able to pass -1 as the `pid`.
126 // * Since it is important to be able to detect -1 has been passed to
127 // `os::waitpid`, as a matter of practicality, we choose to:
128 // (1) Use `long` to represent the `pid` argument.
129 // (2) Disable using any value <= 0 for `pid` on Windows.
130 // * This decision is pragmatic. The reasoning is:
131 // (1) The Windows code paths call `os::waitpid` in only a handful of
132 // places, and in none of these conditions do we need `-1` as a value.
133 // (2) Since PIDs virtually never take on values outside the range of
134 // vanilla signed `long` it is likely that an accidental conversion
135 // will never happen.
136 // (3) Even though it is not formalized in the C specification, the
137 // implementation of `long` on the vast majority of production servers
138 // is 2's complement, so we expect that when we accidentally do
139 // implicitly convert from `unsigned long` to `long`, we will "wrap
140 // around" to negative values. And since we've disabled the negative
141 // `pid` in the Windows implementation, we should error out.
142 inline Result<pid_t> waitpid(pid_t pid, int* status, int options)
143 {
144  const pid_t child_pid = ::waitpid(pid, status, options);
145 
146  if (child_pid == 0) {
147  return None();
148  } else if (child_pid < 0) {
149  return ErrnoError("os::waitpid: Call to `waitpid` failed");
150  } else {
151  return child_pid;
152  }
153 }

220 // Suspends execution for the given duration.
221 inline Try<Nothing> sleep(const Duration& duration)
222 {
223  timespec remaining;
224  remaining.tv_sec = static_cast<long>(duration.secs());
225  remaining.tv_nsec =
226  static_cast<long>((duration - Seconds(remaining.tv_sec)).ns());
227 
228  while (nanosleep(&remaining, &remaining) == -1) {
229  if (errno == EINTR) {
230  continue;
231  } else {
232  return ErrnoError();
233  }
234  }
235 
236  return Nothing();

317 // Overload of os::pids for filtering by groups and sessions.
318 // A group / session id of 0 will fitler on the group / session ID
319 // of the calling process.
320 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
321 {
322  if (group.isNone() && session.isNone()) {
323  return os::pids();
324  } else if (group.isSome() && group.get() < 0) {
325  return Error("Invalid group");
326  } else if (session.isSome() && session.get() < 0) {
327  return Error("Invalid session");
328  }
329 
330  const Try<std::list<Process>> processes = os::processes();
331 
332  if (processes.isError()) {
333  return Error(processes.error());
334  }
335 
336  // Obtain the calling process group / session ID when 0 is provided.
337  if (group.isSome() && group.get() == 0) {
338  group = getpgid(0);
339  }
340  if (session.isSome() && session.get() == 0) {
341  session = getsid(0);
342  }
343 
344  std::set<pid_t> result;
345  foreach (const Process& process, processes.get()) {
346  // Group AND Session (intersection).
347  if (group.isSome() && session.isSome()) {
348  if (group.get() == process.group &&
349  process.session.isSome() &&
350  session.get() == process.session.get()) {
351  result.insert(process.pid);
352  }
353  } else if (group.isSome() && group.get() == process.group) {
354  result.insert(process.pid);
355  } else if (session.isSome() && process.session.isSome() &&
356  session.get() == process.session.get()) {
357  result.insert(process.pid);
358  }
359  }
360 
361  return result;
362 }

379 // Return the OS release numbers.
380 inline Try<Version> release()
381 {
382  Try<UTSInfo> info = uname();
383  if (info.isError()) {
384  return Error(info.error());
385  }
386 
387  int major, minor, patch = 0;
388 #ifndef __FreeBSD__
389  // TODO(karya): Replace sscanf with Version::parse() once Version

392  info.get().release.c_str(),
393  "%d.%d.%d",
394  &major,
395  &minor,
396  &patch) != 3) {
397  return Error("Failed to parse: " + info.get().release);
398  }
399 #else
400  // TODO(dforsyth): Handle FreeBSD patch versions (-pX).
401  if (::sscanf(info.get().release.c_str(), "%d.%d-%*s", &major, &minor) != 2) {
402  return Error("Failed to parse: " + info.get().release);
403  }
404 #endif
405  return Version(major, minor, patch);
406 }

409 inline Option<std::string> which(
410  const std::string& command,
411  const Option<std::string>& _path = None())
412 {
413  Option<std::string> path = _path;
414 
415  if (path.isNone()) {
416  path = getenv("PATH");
417 
418  if (path.isNone()) {
419  return None();
420  }
421  }
422 
423  std::vector<std::string> tokens = strings::tokenize(path.get(), ":");
424  foreach (const std::string& token, tokens) {
425  const std::string commandPath = path::join(token, command);
426  if (!os::exists(commandPath)) {
427  continue;
428  }
429 
430  Try<os::Permissions> permissions = os::permissions(commandPath);
431  if (permissions.isError()) {
432  continue;
433  }
434 
435  if (!permissions.get().owner.x &&
436  !permissions.get().group.x &&
437  !permissions.get().others.x) {
438  continue;
439  }
440 
441  return commandPath;
442  }
443 
444  return None();
445 }
