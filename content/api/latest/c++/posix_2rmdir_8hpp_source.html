32 // By default, recursively deletes a directory akin to: 'rm -r'. If
33 // `recursive` is false, it deletes a directory akin to: 'rmdir'. In
34 // recursive mode, `removeRoot` can be set to false to enable removing
35 // all the files and directories beneath the given root directory, but
36 // not the root directory itself.
37 // Note that this function expects an absolute path.
38 // By default rmdir aborts when an error occurs during the deletion of
39 // any file but if 'continueOnError' is set to true, rmdir logs the error
40 // and continues with the next file.
41 #ifndef __sun // FTS is not available on Solaris.
42 inline Try<Nothing> rmdir(
43  const std::string& directory,
44  bool recursive = true,
45  bool removeRoot = true,
46  bool continueOnError = false)
47 {
48  if (!recursive) {
49  if (::rmdir(directory.c_str()) < 0) {
50  return ErrnoError();
51  }
52  } else {
53  // NOTE: `fts_open` will not always return `nullptr` if the path does not
54  // exist. We manually induce an error here to indicate that we can't remove
55  // a directory that does not exist.
56  if (!os::exists(directory)) {
57  return ErrnoError(ENOENT);
58  }
59 
60  char* paths[] = {const_cast<char*>(directory.c_str()), nullptr};
61 
62  // Using `FTS_PHYSICAL` here because we need `FTSENT` for the
63  // symbolic link in the directory and not the target it links to.
64  FTS* tree = fts_open(paths, (FTS_NOCHDIR | FTS_PHYSICAL), nullptr);
65  if (tree == nullptr) {
66  return ErrnoError();
67  }
68 
69  FTSENT* node;
70  while ((node = fts_read(tree)) != nullptr) {
71  switch (node->fts_info) {
72  case FTS_DP:
73  // Don't remove the root of the traversal of `removeRoot`
74  // is false.
75  if (!removeRoot && node->fts_level == FTS_ROOTLEVEL) {
76  continue;
77  }
78 
79  if (::rmdir(node->fts_path) < 0 && errno != ENOENT) {
80  if (continueOnError) {
81  LOG(ERROR) << "Failed to delete directory "
82  << path::join(directory, node->fts_path)
83  << ": " << os::strerror(errno);
84  } else {
85  Error error = ErrnoError();
86  fts_close(tree);
87  return error;
88  }
89  }
90  break;
91  // `FTS_DEFAULT` would include any file type which is not
92  // explicitly described by any of the other `fts_info` values.
93  case FTS_DEFAULT:
94  case FTS_F:
95  case FTS_SL:
96  // `FTS_SLNONE` should never be the case as we don't set
97  // `FTS_COMFOLLOW` or `FTS_LOGICAL`. Adding here for completion.
98  case FTS_SLNONE:
99  if (::unlink(node->fts_path) < 0 && errno != ENOENT) {
100  if (continueOnError) {
101  LOG(ERROR) << "Failed to delete path "
102  << path::join(directory, node->fts_path)
103  << ": " << os::strerror(errno);
104  } else {
105  Error error = ErrnoError();
106  fts_close(tree);
107  return error;
108  }
109  }
110  break;
111  default:
112  break;
113  }
114  }
115 
116  // If 'continueOnError' is true, we have already logged individual errors.
117  if (errno != 0) {
118  Error error = continueOnError
119  ? Error("rmdir failed in 'continueOnError' mode")
120  : ErrnoError();
121  fts_close(tree);
122  return error;
123  }
124 
125  if (fts_close(tree) < 0) {
126  return ErrnoError();
127  }
128  }
129 
130  return Nothing();
131 }
132 #endif // __sun
