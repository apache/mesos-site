159  void add(const process::UPID& pid)
160  {
161  // Link in order to keep a socket open (more efficient).
162  //
163  // We force a reconnect to avoid sending on a "stale" socket. In
164  // general when linking to a remote process, the underlying TCP
165  // connection may become "stale". RFC 793 refers to this as a
166  // "half-open" connection: the RST is not sent upon the death
167  // of the peer and a RST will only be received once further
168  // data is sent on the socket.
169  //
170  // "Half-open" (aka "stale") connections are typically addressed
171  // via keep-alives (see RFC 1122 4.2.3.6) to periodically probe
172  // the connection. In this case, we can rely on the (re-)addition
173  // of the network member to create a new connection.
174  //
175  // See MESOS-5576 for a scenario where reconnecting helps avoid
176  // dropped messages.
177  link(pid, RemoteConnection::RECONNECT);
178 
179  pids.insert(pid);
180 
181  // Update any pending watches.
182  update();
183  }
184 
185  void remove(const process::UPID& pid)
186  {
187  // TODO(benh): unlink(pid);
188  pids.erase(pid);
189 
190  // Update any pending watches.
191  update();
192  }

205  process::Future<size_t> watch(size_t size, Network::WatchMode mode)

219  // Sends a request to each of the groups members and returns a set
220  // of futures that represent their responses.
221  template <typename Req, typename Res>
222  std::set<process::Future<Res>> broadcast(
223  const Protocol<Req, Res>& protocol,
224  const Req& req,
225  const std::set<process::UPID>& filter)
226  {
227  std::set<process::Future<Res>> futures;
228  typename std::set<process::UPID>::const_iterator iterator;
229  for (iterator = pids.begin(); iterator != pids.end(); ++iterator) {
230  const process::UPID& pid = *iterator;
231  if (filter.count(pid) == 0) {
232  futures.insert(protocol(pid, req));
233  }
234  }
235  return futures;
236  }
237 
238  template <typename M>
239  Nothing broadcast(
240  const M& m,
241  const std::set<process::UPID>& filter)
242  {
243  std::set<process::UPID>::const_iterator iterator;
244  for (iterator = pids.begin(); iterator != pids.end(); ++iterator) {
245  const process::UPID& pid = *iterator;
246  if (filter.count(pid) == 0) {
247  process::post(pid, m);
248  }
249  }
250  return Nothing();
251  }

278  // Notifies the change of the network.
279  void update()
280  {
281  const size_t size = watches.size();
282  for (size_t i = 0; i < size; i++) {
283  Watch* watch = watches.front();
284  watches.pop_front();
285 
286  if (satisfied(watch->size, watch->mode)) {
287  watch->promise.set(pids.size());
288  delete watch;
289  } else {
290  watches.push_back(watch);
291  }
292  }
293  }
294 
295  // Returns true if the current size of the network satisfies the
296  // constraint specified by 'size' and 'mode'.
297  bool satisfied(size_t size, Network::WatchMode mode)
298  {
299  switch (mode) {
300  case Network::EQUAL_TO:
301  return pids.size() == size;
302  case Network::NOT_EQUAL_TO:
303  return pids.size() != size;
304  case Network::LESS_THAN:
305  return pids.size() < size;
306  case Network::LESS_THAN_OR_EQUAL_TO:
307  return pids.size() <= size;
308  case Network::GREATER_THAN:
309  return pids.size() > size;
310  case Network::GREATER_THAN_OR_EQUAL_TO:
311  return pids.size() >= size;
312  default:
313  LOG(FATAL) << "Invalid watch mode";
314  UNREACHABLE();
315  }
316  }
317 
318  std::set<process::UPID> pids;
319  std::list<Watch*> watches;
320 };

419 inline void ZooKeeperNetwork::watched(
420  const process::Future<std::set<zookeeper::Group::Membership>>&)
421 {
422  if (memberships.isFailed()) {
423  // We can't do much here, we could try creating another Group but
424  // that might just continue indefinitely, so we fail early
425  // instead. Note that Group handles all retryable/recoverable
426  // ZooKeeper errors internally.
427  LOG(FATAL) << "Failed to watch ZooKeeper group: " << memberships.failure();
428  }
429 
430  CHECK_READY(memberships); // Not expecting Group to discard futures.
431 
432  LOG(INFO) << "ZooKeeper group memberships changed";
433 
434  // Get data for each membership in order to convert them to PIDs.
435  std::list<process::Future<Option<std::string>>> futures;
436 
437  foreach (const zookeeper::Group::Membership& membership, memberships.get()) {
438  futures.push_back(group.data(membership));
439  }
440 
441  process::collect(futures)
442  .after(Seconds(5),
443  [](process::Future<std::list<Option<std::string>>> datas) {
444  // Handling time outs when collecting membership
445  // data. For now, a timeout is treated as a failure.
446  datas.discard();
447  return process::Failure("Timed out");
448  })
449  .onAny(executor.defer(lambda::bind(&This::collected, this, lambda::_1)));
450 }

453 inline void ZooKeeperNetwork::collected(
454  const process::Future<std::list<Option<std::string>>>& datas)
455 {
456  if (datas.isFailed()) {
457  LOG(WARNING) << "Failed to get data for ZooKeeper group members: "
458  << datas.failure();
459 
460  // Try again later assuming empty group. Note that this does not
461  // remove any of the current group members.
462  watch(std::set<zookeeper::Group::Membership>());
463  return;
464  }
465 
466  CHECK_READY(datas); // Not expecting collect to discard futures.
467 
468  std::set<process::UPID> pids;
469 
470  foreach (const Option<std::string>& data, datas.get()) {
471  // Data could be None if the membership is gone before its
472  // content can be read.
473  if (data.isSome()) {
474  process::UPID pid(data.get());
475  CHECK(pid) << "Failed to parse '" << data.get() << "'";
476  pids.insert(pid);
477  }
478  }
479 
480  LOG(INFO) << "ZooKeeper group PIDs: " << stringify(pids);
481 
482  // Update the network. We make sure that the PIDs from the base set
483  // are always in the network.
484  set(pids | base);
485 
486  watch(memberships.get());
487 }
