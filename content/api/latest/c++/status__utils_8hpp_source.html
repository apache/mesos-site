32 inline std::string WSTRINGIFY(int status)
33 {
34 #ifdef __WINDOWS__
35  // On Windows the exit codes are not standardized. The behaviour should
36  // be defined to improve the diagnostic based on logs.
37  // TODO(dpravat): MESOS-5417 tracks this improvement.
38  LOG(WARNING) << "`WSTRINGIFY` has been called, but it is not implemented.";
39 
40  return "";
41 #else
42  std::string message;
43  if (WIFEXITED(status)) {
44  message += "exited with status ";
45  message += stringify(WEXITSTATUS(status));
46  } else if (WIFSIGNALED(status)) {
47  message += "terminated with signal ";
48  message += strsignal(WTERMSIG(status));
49  if (WCOREDUMP(status)) {
50  message += " (core dumped)";
51  }
52  } else if (WIFSTOPPED(status)) {
53  message += "stopped on signal ";
54  message += strsignal(WSTOPSIG(status));
55  } else {
56  message += "wait status ";
57  message += stringify(status);
58  }
59  return message;
60 #endif // __WINDOWS__
61 }
