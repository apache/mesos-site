38 // Creates a temporary directory using the specified path
39 // template. The template may be any path with _6_ `Xs' appended to
40 // it, for example /tmp/temp.XXXXXX. The trailing `Xs' are replaced
41 // with a unique alphanumeric combination.
42 inline Try<std::string> mkdtemp(
43  const std::string& path = path::join(os::temp(), "XXXXXX"))
44 {
45  // NOTE: We'd like to avoid reallocating `postfixTemplate` and `alphabet`,
46  // and to avoid recomputing their sizes on each call to `mkdtemp`, so we
47  // make them `static const` and use the slightly awkward `sizeof` trick to
48  // compute their sizes once instead of calling `strlen` for each call.
49  static const char postfixTemplate[] = "XXXXXX";
50  static const size_t postfixSize = sizeof(postfixTemplate) - 1;
51 
52  if (!strings::endsWith(path, postfixTemplate)) {
53  return Error(
54  "Invalid template passed to `os::mkdtemp`: template '" + path +
55  "' should end with 6 'X' characters");
56  }
57 
58  static const char alphabet[] =
59  "0123456789"
60  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
61  "abcdefghijklmnopqrstuvwxyz";
62 
63  // NOTE: The maximum addressable index in a string is the total length of the
64  // string minus 1; but C strings have an extra null character at the end, so
65  // the size of the array is actually one more than the length of the string,
66  // which is why we're subtracting 2 here.
67  static const size_t maxAlphabetIndex = sizeof(alphabet) - 2;
68 
69  std::string postfix(postfixTemplate);
70  static THREAD_LOCAL std::mt19937 generator((std::random_device())());
71 
72  for (int i = 0; i < postfixSize; ++i) {
73  int index = generator() % maxAlphabetIndex;
74  postfix[i] = alphabet[index];
75  }
76 
77  // Replace template, make directory.
78  std::string tempPath = path
79  .substr(0, path.length() - postfixSize)
80  .append(postfix);
81 
82  Try<Nothing> mkdir = os::mkdir(tempPath, false);
83 
84  if (mkdir.isError()) {
85  return Error(mkdir.error());
86  }
87 
88  return tempPath;
89 }
