58 inline Try<std::string> nodename()
59 {
60  // MSDN documentation states "The names are established at system startup,
61  // when the system reads them from the registry." This is akin to the
62  // Linux `gethostname` which calls `uname`, thus avoiding a DNS lookup.
63  // The `net::getHostname` function can be used for an explicit DNS lookup.
64  //
65  // NOTE: This returns the hostname of the local computer, or the local
66  // node if this computer is part of a cluster.
67  COMPUTER_NAME_FORMAT format = ComputerNamePhysicalDnsHostname;
68  DWORD size = 0;
69  if (::GetComputerNameExW(format, nullptr, &size) == 0) {
70  if (::GetLastError() != ERROR_MORE_DATA) {
71  return WindowsError();
72  }
73  }
74 
75  std::vector<wchar_t> buffer;
76  buffer.reserve(size);
77 
78  if (::GetComputerNameExW(format, buffer.data(), &size) == 0) {
79  return WindowsError();
80  }
81 
82  return stringify(std::wstring(buffer.data()));
83 }

88 // Overload of os::pids for filtering by groups and sessions. A group / session
89 // id of 0 will fitler on the group / session ID of the calling process.
90 // NOTE: Windows does not have the concept of a process group, so we need to
91 // enumerate all processes.
92 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
93 {
94  DWORD max_items = 4096;
95  DWORD bytes_returned;
96  std::vector<pid_t> processes;
97  size_t size_in_bytes;
98 
99  // Attempt to populate `processes` with PIDs. We repeatedly call
100  // `EnumProcesses` with increasingly large arrays until it "succeeds" at
101  // populating the array with PIDs. The criteria for determining when
102  // `EnumProcesses` has succeeded are:
103  // (1) the return value is nonzero.
104  // (2) the `bytes_returned` is less than the number of bytes in the array.
105  do {
106  // TODO(alexnaparu): Set a limit to the memory that can be used.
107  processes.resize(max_items);
108  size_in_bytes = processes.size() * sizeof(pid_t);
109  CHECK_LE(size_in_bytes, MAXDWORD);
110 
111  BOOL result = ::EnumProcesses(
112  processes.data(),
113  static_cast<DWORD>(size_in_bytes),
114  &bytes_returned);
115 
116  if (!result) {
117  return WindowsError("os::pids: Call to `EnumProcesses` failed");
118  }
119 
120  max_items *= 2;
121  } while (bytes_returned >= size_in_bytes);
122 
123  std::set<pid_t> pids_set(processes.begin(), processes.end());
124 
125  // NOTE: The PID `0` will always be returned by `EnumProcesses`; however, it
126  // is the PID of Windows' System Idle Process. While the PID is valid, using
127  // it for anything is almost always invalid. For instance, `OpenProcess` will
128  // fail with an invalid parameter error if the user tries to get a handle for
129  // PID `0`. In the interest of safety, we prevent the `pids` API from ever
130  // including the PID `0`.
131  pids_set.erase(0);
132  return pids_set;
133 }

184 // Suspends execution of the calling process until a child specified by `pid`
185 // has changed state. Unlike the POSIX standard function `::waitpid`, this
186 // function does not use -1 and 0 to signify errors and nonblocking return.
187 // Instead, we return `Result<pid_t>`:
188 // * In case of error, we return `Error` rather than -1. For example, we
189 // would return an `Error` in case of `EINVAL`.
190 // * In case of nonblocking return, we return `None` rather than 0. For
191 // example, if we pass `WNOHANG` in the `options`, we would expect 0 to be
192 // returned in the case that children specified by `pid` exist, but have
193 // not changed state yet. In this case we return `None` instead.
194 //
195 // NOTE: There are important differences between the POSIX and Windows
196 // implementations of this function:
197 // * On POSIX, `pid_t` is a signed number, but on Windows, PIDs are `DWORD`,
198 // which is `unsigned long`. Thus, if we use `DWORD` to represent the `pid`
199 // argument, passing -1 as the `pid` would (on most modern servers)
200 // silently convert to a really large `pid`. This is undesirable.
201 // * Since it is important to be able to detect -1 has been passed to
202 // `os::waitpid`, as a matter of practicality, we choose to:
203 // (1) Use `long` to represent the `pid` argument.
204 // (2) Disable using any value <= 0 for `pid` on Windows.
205 // * This decision is pragmatic. The reasoning is:
206 // (1) The Windows code paths call `os::waitpid` in only a handful of
207 // places, and in none of these conditions do we need `-1` as a value.
208 // (2) Since PIDs virtually never take on values outside the range of
209 // vanilla signed `long` it is likely that an accidental conversion
210 // will never happen.
211 // (3) Even though it is not formalized in the C specification, the
212 // implementation of `long` on the vast majority of production servers
213 // is 2's complement, so we expect that when we accidentally do
214 // implicitly convert from `unsigned long` to `long`, we will "wrap
215 // around" to negative values. And since we've disabled the negative
216 // `pid` in the Windows implementation, we should error out.
217 // * Finally, on Windows, we currently do not check that the process we are
218 // attempting to await is a child process.
219 inline Result<pid_t> waitpid(long pid, int* status, int options)
220 {
221  const bool wait_for_child = (options & WNOHANG) == 0;
222 
223  // NOTE: Windows does not implement pids <= 0.
224  if (pid <= 0) {
225  errno = ENOSYS;
226  return ErrnoError(
227  "os::waitpid: Value of pid is '" + stringify(pid) +
228  "'; the Windows implementation currently does not allow values <= 0");
229  } else if (options != 0 && options != WNOHANG) {
230  // NOTE: We only support `options == 0` or `options == WNOHANG`. On Windows
231  // no flags other than `WNOHANG` are supported.
232  errno = ENOSYS;
233  return ErrnoError(
234  "os::waitpid: Only flag `WNOHANG` is implemented on Windows");
235  }
236 
237  // TODO(hausdorff): Check that `pid` is one of the child processes. If not,
238  // set `errno` to `ECHILD` and return -1.
239 
240  // Open the child process as a safe `SharedHandle`.
241  const HANDLE process = ::OpenProcess(
242  PROCESS_QUERY_INFORMATION | SYNCHRONIZE,
243  FALSE,
244  static_cast<DWORD>(pid));
245 
246  if (process == nullptr) {
247  return WindowsError("os::waitpid: Failed to open process for pid '" +
248  stringify(pid) + "'");
249  }
250 
251  SharedHandle scoped_process(process, ::CloseHandle);
252 
253  // If `WNOHANG` flag is set, don't wait. Otherwise, wait for child to
254  // terminate.
255  const DWORD wait_time = wait_for_child ? INFINITE : 0;
256  const DWORD wait_results = ::WaitForSingleObject(
257  scoped_process.get(),
258  wait_time);
259 
260  // Verify our wait exited correctly.
261  const bool state_signaled = wait_results == WAIT_OBJECT_0;
262  if (options == 0 && !state_signaled) {
263  // If `WNOHANG` is not set, then we should have stopped waiting only for a
264  // state change in `scoped_process`.
265  errno = ECHILD;
266  return WindowsError(
267  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
268  "'. `::WaitForSingleObject` should have waited for child process to " +
269  "exit, but returned code '" + stringify(wait_results) +
270  "' instead");
271  } else if (wait_for_child && !state_signaled &&
272  wait_results != WAIT_TIMEOUT) {
273  // If `WNOHANG` is set, then a successful wait should report either a
274  // timeout (since we set the time to wait to `0`), or a successful state
275  // change of `scoped_process`. Anything else is an error.
276  errno = ECHILD;
277  return WindowsError(
278  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
279  "'. `ENOHANG` flag was passed in, so `::WaitForSingleObject` should " +
280  "have either returned `WAIT_OBJECT_0` or `WAIT_TIMEOUT` (the " +
281  "timeout was set to 0, because we are not waiting for the child), " +
282  "but instead returned code '" + stringify(wait_results) + "'");
283  }
284 
285  if (!wait_for_child && wait_results == WAIT_TIMEOUT) {
286  // Success. `ENOHANG` was set and we got a timeout, so return `None` (POSIX
287  // `::waitpid` would return 0 here).
288  return None();
289  }
290 
291  // Attempt to retrieve exit code from child process. Store that exit code in
292  // the `status` variable if it's `nullptr`.
293  DWORD child_exit_code = 0;
294  if (!::GetExitCodeProcess(scoped_process.get(), &child_exit_code)) {
295  errno = ECHILD;
296  return WindowsError(
297  "os::waitpid: Successfully waited on child process with pid '" +
298  std::to_string(pid) + "', but could not retrieve exit code");
299  }
300 
301  if (status != nullptr) {
302  *status = child_exit_code;
303  }
304 
305  // Success. Return pid of the child process for which the status is reported.
306  return pid;
307 }

402 inline Result<PROCESSENTRY32W> process_entry(pid_t pid)
403 {
404  // Get a snapshot of the processes in the system. NOTE: We should not check
405  // whether the handle is `nullptr`, because this API will always return
406  // `INVALID_HANDLE_VALUE` on error.
407  HANDLE snapshot_handle = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, pid);
408  if (snapshot_handle == INVALID_HANDLE_VALUE) {
409  return WindowsError(
410  "os::process_entry: Call to `CreateToolhelp32Snapshot` failed");
411  }
412 
413  SharedHandle safe_snapshot_handle(snapshot_handle, ::CloseHandle);
414 
415  // Initialize process entry.
416  PROCESSENTRY32W process_entry;
417  memset(&process_entry, 0, sizeof(process_entry));
418  process_entry.dwSize = sizeof(process_entry);
419 
420  // Get first process so that we can loop through process entries until we
421  // find the one we care about.
422  SetLastError(ERROR_SUCCESS);
423  BOOL has_next = Process32First(safe_snapshot_handle.get(), &process_entry);
424  if (has_next == FALSE) {
425  // No first process was found. We should never be here; it is arguable we
426  // should return `None`, since we won't find the PID we're looking for, but
427  // we elect to return `Error` because something terrible has probably
428  // happened.
429  if (::GetLastError() != ERROR_SUCCESS) {
430  return WindowsError("os::process_entry: Call to `Process32First` failed");
431  } else {
432  return Error("os::process_entry: Call to `Process32First` failed");
433  }
434  }
435 
436  // Loop through processes until we find the one we're looking for.
437  while (has_next == TRUE) {
438  if (process_entry.th32ProcessID == pid) {
439  // Process found.
440  return process_entry;
441  }
442 
443  has_next = Process32Next(safe_snapshot_handle.get(), &process_entry);
444  if (has_next == FALSE) {
445  DWORD last_error = ::GetLastError();
446  if (last_error != ERROR_NO_MORE_FILES && last_error != ERROR_SUCCESS) {
447  return WindowsError(
448  "os::process_entry: Call to `Process32Next` failed");
449  }
450  }
451  }
452 
453  return None();
454 }

457 // Generate a `Process` object for the process associated with `pid`. If
458 // process is not found, we return `None`; error is reserved for the case where
459 // something went wrong.
460 inline Result<Process> process(pid_t pid)
461 {
462  if (pid == 0) {
463  // The 0th PID is that of the System Idle Process on Windows. However, it is
464  // invalid to attempt to get a proces handle or else perform any operation
465  // on this pseudo-process.
466  return Error("os::process: Invalid parameter: pid == 0");
467  }
468 
469  // Find process with pid.
470  Result<PROCESSENTRY32W> entry = process_entry(pid);
471 
472  if (entry.isError()) {
473  return WindowsError(entry.error());
474  } else if (entry.isNone()) {
475  return None();
476  }
477 
478  HANDLE process_handle = ::OpenProcess(
479  PROCESS_QUERY_LIMITED_INFORMATION | PROCESS_VM_READ,
480  false,
481  pid);
482 
483  // ::OpenProcess returns `NULL`, not `INVALID_HANDLE_VALUE` on failure.
484  if (process_handle == nullptr) {
485  return WindowsError("os::process: Call to `OpenProcess` failed");
486  }
487 
488  SharedHandle safe_process_handle(process_handle, ::CloseHandle);
489 
490  // Get Windows Working set size (Resident set size in linux).
491  PROCESS_MEMORY_COUNTERS proc_mem_counters;
492  BOOL get_process_memory_info = ::GetProcessMemoryInfo(
493  safe_process_handle.get_handle(),
494  &proc_mem_counters,
495  sizeof(proc_mem_counters));
496 
497  if (!get_process_memory_info) {
498  return WindowsError("os::process: Call to `GetProcessMemoryInfo` failed");
499  }
500 
501  // Get session Id.
502  pid_t session_id;
503  BOOL process_id_to_session_id = ::ProcessIdToSessionId(pid, &session_id);
504 
505  if (!process_id_to_session_id) {
506  return WindowsError("os::process: Call to `ProcessIdToSessionId` failed");
507  }
508 
509  // Get Process CPU time.
510  FILETIME create_filetime, exit_filetime, kernel_filetime, user_filetime;
511  BOOL get_process_times = ::GetProcessTimes(
512  safe_process_handle.get_handle(),
513  &create_filetime,
514  &exit_filetime,
515  &kernel_filetime,
516  &user_filetime);
517 
518  if (!get_process_times) {
519  return WindowsError("os::process: Call to `GetProcessTimes` failed");
520  }
521 
522  // Get utime and stime.
523  ULARGE_INTEGER lKernelTime, lUserTime; // In 100 nanoseconds.
524  lKernelTime.HighPart = kernel_filetime.dwHighDateTime;
525  lKernelTime.LowPart = kernel_filetime.dwLowDateTime;
526  lUserTime.HighPart = user_filetime.dwHighDateTime;
527  lUserTime.LowPart = user_filetime.dwLowDateTime;
528 
529  Try<Duration> utime = Nanoseconds(lKernelTime.QuadPart * 100);
530  Try<Duration> stime = Nanoseconds(lUserTime.QuadPart * 100);
531 
532  return Process(
533  pid,
534  entry.get().th32ParentProcessID, // Parent process id.
535  0, // Group id.
536  session_id,
537  Bytes(proc_mem_counters.WorkingSetSize),
538  utime.isSome() ? utime.get() : Option<Duration>::none(),
539  stime.isSome() ? stime.get() : Option<Duration>::none(),
540  stringify(entry.get().szExeFile), // Executable filename.
541  false); // Is not zombie process.
542 }

606 // `create_job` function creates a named job object using `name`.
607 // This returns the safe job handle, which closes the job handle
608 // when destructed. Because the job is destroyed when its last
609 // handle is closed and all associated processes have exited,
610 // a running process must be assigned to the created job
611 // before the returned handle is closed.
612 inline Try<SharedHandle> create_job(const std::wstring& name)
613 {
614  SharedHandle job_handle(
615  ::CreateJobObjectW(
616  nullptr, // Use a default security descriptor, and
617  // the created handle cannot be inherited.
618  name.data()), // The name of the job.
619  ::CloseHandle);
620 
621  if (job_handle.get_handle() == nullptr) {
622  return WindowsError(
623  "os::create_job: Call to `CreateJobObject` failed for job: " +
624  stringify(name));
625  }
626 
627  JOBOBJECT_EXTENDED_LIMIT_INFORMATION info = {};
628 
629  // The job object will be terminated when the job handle closes. This allows
630  // the job tree to be terminated in case of errors by closing the handle.
631  // We set this flag so that the death of the agent process will
632  // always kill any running jobs, as the OS will close the remaining open
633  // handles if all destructors failed to run (catastrophic death).

685 Result<std::set<Process>> _get_job_processes(const SharedHandle& job_handle) {
686  // This is a statically allocated `JOBOBJECT_BASIC_PROCESS_ID_LIST`. We lie to
687  // the Windows API and construct our own struct to avoid (a) having to do
688  // hairy size calculations and (b) having to allocate dynamically, and then
689  // worry about deallocating.
690  struct {
691  DWORD NumberOfAssignedProcesses;
692  DWORD NumberOfProcessIdsInList;
693  DWORD ProcessIdList[max_pids];
694  } pid_list;
695 
696  BOOL result = ::QueryInformationJobObject(
697  job_handle.get_handle(),
698  JobObjectBasicProcessIdList,
699  reinterpret_cast<JOBOBJECT_BASIC_PROCESS_ID_LIST*>(&pid_list),
700  sizeof(pid_list),
701  nullptr);
702 
703  // `ERROR_MORE_DATA` indicates we need a larger `max_pids`.
704  if (result == FALSE && ::GetLastError() == ERROR_MORE_DATA) {
705  return None();
706  }
707 
708  if (result == FALSE) {
709  return WindowsError(
710  "os::_get_job_processes: call to `QueryInformationJobObject` failed");
711  }

714  for (DWORD i = 0; i < pid_list.NumberOfProcessIdsInList; ++i) {
715  Result<Process> process = os::process(pid_list.ProcessIdList[i]);
716  if (process.isSome()) {
717  processes.insert(process.get());
718  }
719  }
720 
721  return processes;

786 // `set_job_cpu_limit` sets a CPU limit for the process represented by
787 // `pid`, assuming it is assigned to a job object. This function will fail
788 // otherwise. This limit is a hard cap enforced by the OS.
789 //
790 // https://msdn.microsoft.com/en-us/library/windows/desktop/hh448384(v=vs.85).aspx // NOLINT(whitespace/line_length)
791 inline Try<Nothing> set_job_cpu_limit(pid_t pid, double cpus)
792 {
793  JOBOBJECT_CPU_RATE_CONTROL_INFORMATION control_info = {};
794  control_info.ControlFlags =
795  JOB_OBJECT_CPU_RATE_CONTROL_ENABLE |
796  JOB_OBJECT_CPU_RATE_CONTROL_HARD_CAP;
797 
798  // This `CpuRate` is the number of cycles per 10,000 cycles, or a percentage
799  // times 100, e.g. 20% yields 20 * 100 = 2,000. However, the `cpus` argument
800  // represents 1 CPU core with `1.0`, so a 100% CPU limit on a quad-core
801  // machine would be `4.0 cpus`. Thus a mapping of `cpus` to `CpuRate` is
802  // `(cpus / os::cpus()) * 100 * 100`, or the requested `cpus` divided by the
803  // number of CPUs to obtain a fractional representation, multiplied by 100 to
804  // make it a percentage, multiplied again by 100 to become a `CpuRate`.
805  Try<long> total_cpus = os::cpus();
806  control_info.CpuRate =
807  static_cast<DWORD>((cpus / total_cpus.get()) * 100 * 100);
808  // This must not be set to 0, so 1 is the minimum.
809  if (control_info.CpuRate < 1) {
810  control_info.CpuRate = 1;
811  }
812 
813  Try<SharedHandle> job_handle = os::open_job(
814  JOB_OBJECT_SET_ATTRIBUTES,
815  false,
816  pid);
817  if (job_handle.isError()) {
818  return Error(job_handle.error());
819  }
820 
821  BOOL result = ::SetInformationJobObject(
822  job_handle.get().get_handle(),
823  JobObjectCpuRateControlInformation,
824  &control_info,
825  sizeof(control_info));
826  if (result == FALSE) {
827  return WindowsError(
828  "os::set_job_cpu_limit: call to `SetInformationJobObject` failed");
829  }
830 
831  return Nothing();
832 }
