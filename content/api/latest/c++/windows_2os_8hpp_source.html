145 // Overload of os::pids for filtering by groups and sessions. A group / session
146 // id of 0 will fitler on the group / session ID of the calling process.
147 // NOTE: Windows does not have the concept of a process group, so we need to
148 // enumerate all processes.
149 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
150 {
151  DWORD max_items = 4096;
152  DWORD bytes_returned;
153  std::vector<pid_t> processes;
154  size_t size_in_bytes;
155 
156  // Attempt to populate `processes` with PIDs. We repeatedly call
157  // `EnumProcesses` with increasingly large arrays until it "succeeds" at
158  // populating the array with PIDs. The criteria for determining when
159  // `EnumProcesses` has succeeded are:
160  // (1) the return value is nonzero.
161  // (2) the `bytes_returned` is less than the number of bytes in the array.
162  do {
163  // TODO(alexnaparu): Set a limit to the memory that can be used.
164  processes.resize(max_items);
165  size_in_bytes = processes.size() * sizeof(pid_t);
166  CHECK_LE(size_in_bytes, MAXDWORD);
167 
168  BOOL result = ::EnumProcesses(
169  processes.data(),
170  static_cast<DWORD>(size_in_bytes),
171  &bytes_returned);
172 
173  if (!result) {
174  return WindowsError("os::pids: Call to `EnumProcesses` failed");
175  }
176 
177  max_items *= 2;
178  } while (bytes_returned >= size_in_bytes);
179 
180  return std::set<pid_t>(processes.begin(), processes.end());
181 }

206  return;

208

210  ::SetEnvironmentVariable(key.c_str(), value.c_str());

224 // Suspends execution of the calling process until a child specified by `pid`
225 // has changed state. Unlike the POSIX standard function `::waitpid`, this
226 // function does not use -1 and 0 to signify errors and nonblocking return.
227 // Instead, we return `Result<pid_t>`:
228 // * In case of error, we return `Error` rather than -1. For example, we
229 // would return an `Error` in case of `EINVAL`.
230 // * In case of nonblocking return, we return `None` rather than 0. For
231 // example, if we pass `WNOHANG` in the `options`, we would expect 0 to be
232 // returned in the case that children specified by `pid` exist, but have
233 // not changed state yet. In this case we return `None` instead.
234 //
235 // NOTE: There are important differences between the POSIX and Windows
236 // implementations of this function:
237 // * On POSIX, `pid_t` is a signed number, but on Windows, PIDs are `DWORD`,
238 // which is `unsigned long`. Thus, if we use `DWORD` to represent the `pid`
239 // argument, passing -1 as the `pid` would (on most modern servers)
240 // silently convert to a really large `pid`. This is undesirable.
241 // * Since it is important to be able to detect -1 has been passed to
242 // `os::waitpid`, as a matter of practicality, we choose to:
243 // (1) Use `long` to represent the `pid` argument.
244 // (2) Disable using any value <= 0 for `pid` on Windows.
245 // * This decision is pragmatic. The reasoning is:
246 // (1) The Windows code paths call `os::waitpid` in only a handful of
247 // places, and in none of these conditions do we need `-1` as a value.
248 // (2) Since PIDs virtually never take on values outside the range of
249 // vanilla signed `long` it is likely that an accidental conversion
250 // will never happen.
251 // (3) Even though it is not formalized in the C specification, the
252 // implementation of `long` on the vast majority of production servers
253 // is 2's complement, so we expect that when we accidentally do
254 // implicitly convert from `unsigned long` to `long`, we will "wrap
255 // around" to negative values. And since we've disabled the negative
256 // `pid` in the Windows implementation, we should error out.
257 // * Finally, on Windows, we currently do not check that the process we are
258 // attempting to await is a child process.
259 inline Result<pid_t> waitpid(long pid, int* status, int options)
260 {
261  const bool wait_for_child = (options & WNOHANG) == 0;
262 
263  // NOTE: Windows does not implement pids <= 0.
264  if (pid <= 0) {
265  errno = ENOSYS;
266  return ErrnoError(
267  "os::waitpid: Value of pid is '" + stringify(pid) +
268  "'; the Windows implementation currently does not allow values <= 0");
269  } else if (options != 0 && options != WNOHANG) {
270  // NOTE: We only support `options == 0` or `options == WNOHANG`. On Windows
271  // no flags other than `WNOHANG` are supported.
272  errno = ENOSYS;
273  return ErrnoError(
274  "os::waitpid: Only flag `WNOHANG` is implemented on Windows");
275  }
276 
277  // TODO(hausdorff): Check that `pid` is one of the child processes. If not,
278  // set `errno` to `ECHILD` and return -1.
279 
280  // Open the child process as a safe `SharedHandle`.
281  const HANDLE process = ::OpenProcess(
282  PROCESS_QUERY_INFORMATION | SYNCHRONIZE,
283  FALSE,
284  static_cast<DWORD>(pid));
285 
286  if (process == nullptr) {
287  return WindowsError("os::waitpid: Failed to open process for pid '" +
288  stringify(pid) + "'");
289  }
290 
291  SharedHandle scoped_process(process, ::CloseHandle);
292 
293  // If `WNOHANG` flag is set, don't wait. Otherwise, wait for child to
294  // terminate.
295  const DWORD wait_time = wait_for_child ? INFINITE : 0;
296  const DWORD wait_results = ::WaitForSingleObject(
297  scoped_process.get(),
298  wait_time);
299 
300  // Verify our wait exited correctly.
301  const bool state_signaled = wait_results == WAIT_OBJECT_0;
302  if (options == 0 && !state_signaled) {
303  // If `WNOHANG` is not set, then we should have stopped waiting only for a
304  // state change in `scoped_process`.
305  errno = ECHILD;
306  return WindowsError(
307  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
308  "'. `::WaitForSingleObject` should have waited for child process to " +
309  "exit, but returned code '" + stringify(wait_results) +
310  "' instead");
311  } else if (wait_for_child && !state_signaled &&
312  wait_results != WAIT_TIMEOUT) {
313  // If `WNOHANG` is set, then a successful wait should report either a
314  // timeout (since we set the time to wait to `0`), or a successful state
315  // change of `scoped_process`. Anything else is an error.
316  errno = ECHILD;
317  return WindowsError(
318  "os::waitpid: Failed to wait for pid '" + stringify(pid) +
319  "'. `ENOHANG` flag was passed in, so `::WaitForSingleObject` should " +
320  "have either returned `WAIT_OBJECT_0` or `WAIT_TIMEOUT` (the " +
321  "timeout was set to 0, because we are not waiting for the child), " +
322  "but instead returned code '" + stringify(wait_results) + "'");
323  }
324 
325  if (!wait_for_child && wait_results == WAIT_TIMEOUT) {
326  // Success. `ENOHANG` was set and we got a timeout, so return `None` (POSIX
327  // `::waitpid` would return 0 here).
328  return None();
329  }
330 
331  // Attempt to retrieve exit code from child process. Store that exit code in
332  // the `status` variable if it's `nullptr`.
333  DWORD child_exit_code = 0;
334  if (!::GetExitCodeProcess(scoped_process.get(), &child_exit_code)) {
335  errno = ECHILD;
336  return WindowsError(
337  "os::waitpid: Successfully waited on child process with pid '" +
338  std::to_string(pid) + "', but could not retrieve exit code");
339  }
340 
341  if (status != nullptr) {
342  *status = child_exit_code;
343  }
344 
345  // Success. Return pid of the child process for which the status is reported.
346  return pid;
347 }

350 inline std::string hstrerror(int err)
351 {
352  char buffer[1024];
353  DWORD format_error = 0;
354 
355  // NOTE: Per the Linux documentation[1], `h_errno` can have only one of the
356  // following errors.
357  switch (err) {
358  case WSAHOST_NOT_FOUND:
359  case WSANO_DATA:
360  case WSANO_RECOVERY:
361  case WSATRY_AGAIN: {
362  format_error = ::FormatMessage(
363  FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
364  nullptr,
365  err,
366  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
367  buffer,
368  sizeof(buffer),
369  nullptr);
370  break;
371  }
372  default: {
373  return "Unknown resolver error";
374  }
375  }
376 
377  if (format_error == 0) {
378  // If call to `FormatMessage` fails, then we choose to output the error
379  // code rather than call `FormatMessage` again.
380  return "os::hstrerror: Call to `FormatMessage` failed with error code" +
381  std::to_string(GetLastError());
382  } else {
383  return buffer;
384  }
385 }

515 inline Result<PROCESSENTRY32> process_entry(pid_t pid)
516 {
517  // Get a snapshot of the processes in the system. NOTE: We should not check
518  // whether the handle is `nullptr`, because this API will always return
519  // `INVALID_HANDLE_VALUE` on error.
520  HANDLE snapshot_handle = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, pid);
521  if (snapshot_handle == INVALID_HANDLE_VALUE) {
522  return WindowsError(
523  "os::process_entry: Call to `CreateToolhelp32Snapshot` failed");
524  }
525 
526  SharedHandle safe_snapshot_handle(snapshot_handle, ::CloseHandle);
527 
528  // Initialize process entry.
529  PROCESSENTRY32 process_entry;
530  ZeroMemory(&process_entry, sizeof(PROCESSENTRY32));
531  process_entry.dwSize = sizeof(PROCESSENTRY32);
532 
533  // Get first process so that we can loop through process entries until we
534  // find the one we care about.
535  SetLastError(ERROR_SUCCESS);
536  BOOL has_next = Process32First(safe_snapshot_handle.get(), &process_entry);
537  if (has_next == FALSE) {
538  // No first process was found. We should never be here; it is arguable we
539  // should return `None`, since we won't find the PID we're looking for, but
540  // we elect to return `Error` because something terrible has probably
541  // happened.
542  if (GetLastError() != ERROR_SUCCESS) {
543  return WindowsError("os::process_entry: Call to `Process32First` failed");
544  } else {
545  return Error("os::process_entry: Call to `Process32First` failed");
546  }
547  }
548 
549  // Loop through processes until we find the one we're looking for.
550  while (has_next == TRUE) {
551  if (process_entry.th32ProcessID == pid) {
552  // Process found.
553  return process_entry;
554  }
555 
556  has_next = Process32Next(safe_snapshot_handle.get(), &process_entry);
557  if (has_next == FALSE) {
558  DWORD last_error = GetLastError();
559  if (last_error != ERROR_NO_MORE_FILES && last_error != ERROR_SUCCESS) {
560  return WindowsError(
561  "os::process_entry: Call to `Process32Next` failed");
562  }
563  }
564  }
565 
566  return None();
567 }

570 // Generate a `Process` object for the process associated with `pid`. If
571 // process is not found, we return `None`; error is reserved for the case where
572 // something went wrong.
573 inline Result<Process> process(pid_t pid)
574 {
575  // Find process with pid.
576  Result<PROCESSENTRY32> entry = process_entry(pid);
577 
578  if (entry.isError()) {
579  return WindowsError(entry.error());
580  } else if (entry.isNone()) {
581  return None();
582  }
583 
584  HANDLE process_handle = ::OpenProcess(
585  PROCESS_QUERY_LIMITED_INFORMATION | PROCESS_VM_READ,
586  false,
587  pid);
588 
589  if (process_handle == INVALID_HANDLE_VALUE) {
590  return WindowsError("os::process: Call to `OpenProcess` failed");
591  }
592 
593  SharedHandle safe_process_handle(process_handle, ::CloseHandle);
594 
595  // Get Windows Working set size (Resident set size in linux).
596  PROCESS_MEMORY_COUNTERS proc_mem_counters;
597  BOOL get_process_memory_info = ::GetProcessMemoryInfo(
598  safe_process_handle.get_handle(),
599  &proc_mem_counters,
600  sizeof(proc_mem_counters));
601 
602  if (!get_process_memory_info) {
603  return WindowsError("os::process: Call to `GetProcessMemoryInfo` failed");
604  }
605 
606  // Get session Id.
607  pid_t session_id;
608  BOOL process_id_to_session_id = ::ProcessIdToSessionId(pid, &session_id);
609 
610  if (!process_id_to_session_id) {
611  return WindowsError("os::process: Call to `ProcessIdToSessionId` failed");
612  }
613 
614  // Get Process CPU time.
615  FILETIME create_filetime, exit_filetime, kernel_filetime, user_filetime;
616  BOOL get_process_times = ::GetProcessTimes(
617  safe_process_handle.get_handle(),
618  &create_filetime,
619  &exit_filetime,
620  &kernel_filetime,
621  &user_filetime);
622 
623  if (!get_process_times) {
624  return WindowsError("os::process: Call to `GetProcessTimes` failed");
625  }
626 
627  // Get utime and stime.
628  ULARGE_INTEGER lKernelTime, lUserTime; // In 100 nanoseconds.
629  lKernelTime.HighPart = kernel_filetime.dwHighDateTime;
630  lKernelTime.LowPart = kernel_filetime.dwLowDateTime;
631  lUserTime.HighPart = user_filetime.dwHighDateTime;
632  lUserTime.LowPart = user_filetime.dwLowDateTime;
633 
634  Try<Duration> utime = Nanoseconds(lKernelTime.QuadPart * 100);
635  Try<Duration> stime = Nanoseconds(lUserTime.QuadPart * 100);
636 
637  return Process(
638  pid,
639  entry.get().th32ParentProcessID, // Parent process id.
640  0, // Group id.
641  session_id,
642  Bytes(proc_mem_counters.WorkingSetSize),
643  utime.isSome() ? utime.get() : Option<Duration>::none(),
644  stime.isSome() ? stime.get() : Option<Duration>::none(),
645  entry.get().szExeFile, // Executable filename.
646  false); // Is not zombie process.
647 }

660 // the name used in `kill_job`.
661 inline Try<HANDLE> create_job(pid_t pid)
662 {
663  Try<std::string> alpha_pid = strings::internal::format("MESOS_JOB_%X", pid);
664  if (alpha_pid.isError()) {
665  return Error(alpha_pid.error());
666  }
667 
668  HANDLE process_handle = ::OpenProcess(
669  PROCESS_SET_QUOTA | PROCESS_TERMINATE,
670  false,
671  pid);
672 
673  if (process_handle == INVALID_HANDLE_VALUE) {
674  return WindowsError("os::create_job: Call to `OpenProcess` failed");
675  }
676 
677  SharedHandle safe_process_handle(process_handle, ::CloseHandle);
678 
679  HANDLE job_handle = ::CreateJobObject(nullptr, alpha_pid.get().c_str());
680 
681  if (job_handle == nullptr) {
682  return WindowsError("os::create_job: Call to `CreateJobObject` failed");
683  }
684 
685  JOBOBJECT_EXTENDED_LIMIT_INFORMATION jeli = { { 0 }, 0 };
686 
687  jeli.BasicLimitInformation.LimitFlags = JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE;
688 
689  // The job object will be terminated when the job handle closes. This allows
690  // the job tree to be terminated in case of errors by closing the handle.
691  ::SetInformationJobObject(
692  job_handle,
693  JobObjectExtendedLimitInformation,
694  &jeli,
695  sizeof(jeli));
696 
697  if (::AssignProcessToJobObject(
698  job_handle,
699  safe_process_handle.get_handle()) == 0) {
700  return WindowsError(
701  "os::create_job: Call to `AssignProcessToJobObject` failed");
702  };
703 
704  return job_handle;
705 }

708 // `kill_job` function assumes the process identified by `pid`
709 // is associated with a job object whose name is derived from it.
710 // Every process started by the `pid` process which is part of the job
711 // object becomes part of the job object. Destroying the task
712 // will close all such processes.
713 inline Try<Nothing> kill_job(pid_t pid)
714 {
715  Try<std::string> alpha_pid = strings::internal::format("MESOS_JOB_%X", pid);
716  if (alpha_pid.isError()) {
717  return Error(alpha_pid.error());
718  }
719 
720  HANDLE job_handle = ::OpenJobObject(
721  JOB_OBJECT_TERMINATE,
722  FALSE,
723  alpha_pid.get().c_str());
724 
725  if (job_handle == nullptr) {
726  return WindowsError("os::kill_job: Call to `OpenJobObject` failed");
727  }
728 
729  SharedHandle safe_job_handle(job_handle, ::CloseHandle);
730 
731  BOOL result = ::TerminateJobObject(safe_job_handle.get_handle(), 1);
732  if (result == 0) {
733  return WindowsError();
734  }
735 
736  return Nothing();
737 }

740 inline Try<std::string> var()

742  // Get the `ProgramData` path. First, find the size of the output buffer.
743  // This size includes the null-terminating character.
744  DWORD size = 0;

746  // The expected behavior here is for the function to "fail"
747  // and return `false`, and `size` receives necessary buffer size.
748  return WindowsError(
749  "os::var: `GetAllUsersProfileDirectory` succeeded unexpectedly");
750  }
751

753  if (!::GetAllUsersProfileDirectoryW(&var_folder[0], &size)) {
754  return WindowsError(

756  }
757 
758  // Convert UTF-16 `wchar[]` to UTF-8 `string`.
759  std::wstring wvar_folder(&var_folder[0]);
760  std::wstring_convert<std::codecvt_utf8_utf16<wchar_t>, wchar_t> converter;
761  return converter.to_bytes(wvar_folder);
