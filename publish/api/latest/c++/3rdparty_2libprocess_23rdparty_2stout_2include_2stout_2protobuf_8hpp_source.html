58 // Write out the given protobuf to the specified file descriptor by
59 // first writing out the length of the protobuf followed by the
60 // contents.
61 // NOTE: On error, this may have written partial data to the file.
62 inline Try<Nothing> write(int fd, const google::protobuf::Message& message)
63 {
64  if (!message.IsInitialized()) {
65  return Error(message.InitializationErrorString() +
66  " is required but not initialized");
67  }
68 
69  // First write the size of the protobuf.
70  uint32_t size = message.ByteSize();
71  std::string bytes((char*) &size, sizeof(size));
72 
73  Try<Nothing> result = os::write(fd, bytes);
74  if (result.isError()) {
75  return Error("Failed to write size: " + result.error());
76  }
77 
78  if (!message.SerializeToFileDescriptor(fd)) {
79  return Error("Failed to write/serialize message");
80  }
81 
82  return Nothing();
83 }

87 // specified file descriptor by repeatedly invoking write
88 // on each of the messages.

90 template <typename T>
91 Try<Nothing> write(
92  int fd,
93  const google::protobuf::RepeatedPtrField<T>& messages)
94 {
95  foreach (const T& message, messages) {
96  Try<Nothing> result = write(fd, message);
97  if (result.isError()) {
98  return Error(result.error());
99  }
100  }
101 
102  return Nothing();

155 // Reads a single message of type T from the file by first reading the
156 // "size" followed by the contents (as written by 'write' above).
157 // NOTE: This struct is used by the public 'read' function.
158 // See comments there for the reason why we need this.
159 template <typename T>
160 struct Read
161 {
162  Result<T> operator()(int fd, bool ignorePartial, bool undoFailed)
163  {
164  off_t offset = 0;
165 
166  if (undoFailed) {
167  // Save the offset so we can re-adjust if something goes wrong.
168  offset = lseek(fd, 0, SEEK_CUR);
169  if (offset == -1) {
170  return ErrnoError("Failed to lseek to SEEK_CUR");
171  }
172  }
173 
174  uint32_t size;
175  Result<std::string> result = os::read(fd, sizeof(size));
176 
177  if (result.isError()) {
178  if (undoFailed) {
179  lseek(fd, offset, SEEK_SET);
180  }
181  return Error("Failed to read size: " + result.error());
182  } else if (result.isNone()) {
183  return None(); // No more protobufs to read.
184  } else if (result.get().size() < sizeof(size)) {
185  // Hit EOF unexpectedly.
186  if (undoFailed) {
187  // Restore the offset to before the size read.
188  lseek(fd, offset, SEEK_SET);
189  }
190  if (ignorePartial) {
191  return None();
192  }
193  return Error(
194  "Failed to read size: hit EOF unexpectedly, possible corruption");
195  }
196 
197  // Parse the size from the bytes.
198  memcpy((void*) &size, (void*) result.get().data(), sizeof(size));
199 
200  // NOTE: Instead of specifically checking for corruption in 'size',
201  // we simply try to read 'size' bytes. If we hit EOF early, it is an
202  // indication of corruption.
203  result = os::read(fd, size);
204 
205  if (result.isError()) {
206  if (undoFailed) {
207  // Restore the offset to before the size read.
208  lseek(fd, offset, SEEK_SET);
209  }
210  return Error("Failed to read message: " + result.error());
211  } else if (result.isNone() || result.get().size() < size) {
212  // Hit EOF unexpectedly.
213  if (undoFailed) {
214  // Restore the offset to before the size read.
215  lseek(fd, offset, SEEK_SET);
216  }
217  if (ignorePartial) {
218  return None();
219  }
220  return Error("Failed to read message of size " + stringify(size) +
221  " bytes: hit EOF unexpectedly, possible corruption");
222  }
223 
224  // Parse the protobuf from the string.
225  // NOTE: We need to capture a const reference to the data because it
226  // must outlive the creation of ArrayInputStream.
227  const std::string& data = result.get();
228 
229  T message;
230  google::protobuf::io::ArrayInputStream stream(data.data(), data.size());
231 
232  if (!message.ParseFromZeroCopyStream(&stream)) {
233  if (undoFailed) {
234  // Restore the offset to before the size read.
235  lseek(fd, offset, SEEK_SET);
236  }
237  return Error("Failed to deserialize message");
238  }
239 
240  return message;
241  }
242 };

245 // Partial specialization for RepeatedPtrField<T> to read a sequence
246 // of protobuf messages from a given fd by repeatedly invoking
247 // Read<T> until None is reached, which we treat as EOF.
248 // NOTE: This struct is used by the public 'read' function.
249 // See comments there for the reason why we need this.
250 template <typename T>
251 struct Read<google::protobuf::RepeatedPtrField<T>>
252 {
253  Result<google::protobuf::RepeatedPtrField<T>> operator()(
254  int fd, bool ignorePartial, bool undoFailed)
255  {
256  google::protobuf::RepeatedPtrField<T> result;
257  for (;;) {
258  Result<T> message = Read<T>()(fd, ignorePartial, undoFailed);
259  if (message.isError()) {
260  return Error(message.error());
261  } else if (message.isNone()) {
262  break;
263  } else {
264  result.Add()->CopyFrom(message.get());
265  }
266  }
267  return result;
268  }
269 };

274 // Reads the protobuf message(s) from a given fd based on the format
275 // written by write() above. We use partial specialization of
276 // - internal::Read<T> vs
277 // - internal::Read<google::protobuf::RepeatedPtrField<T>>
278 // in order to determine whether T is a single protobuf message or
279 // a sequence of messages.
280 // If 'ignorePartial' is true, None() is returned when we unexpectedly
281 // hit EOF while reading the protobuf (e.g., partial write).
282 // If 'undoFailed' is true, failed read attempts will restore the file
283 // read/write file offset towards the initial callup position.
284 template <typename T>
285 Result<T> read(int fd, bool ignorePartial = false, bool undoFailed = false)
286 {
287  return internal::Read<T>()(fd, ignorePartial, undoFailed);
288 }

324 struct Parser : boost::static_visitor<Try<Nothing> >
325 {
326  Parser(google::protobuf::Message* _message,
327  const google::protobuf::FieldDescriptor* _field)
328  : message(_message),
329  reflection(message->GetReflection()),
330  field(_field) {}
331 
332  Try<Nothing> operator()(const JSON::Object& object) const
333  {
334  switch (field->type()) {
335  case google::protobuf::FieldDescriptor::TYPE_MESSAGE:
336  // TODO(gilbert): We currently push up the nested error
337  // messages without wrapping the error message (due to
338  // the recursive nature of parse). We should pass along
339  // variable information in order to construct a helpful
340  // error message, e.g. "Failed to parse field 'a.b.c': ...".
341  if (field->is_repeated()) {
342  return parse(reflection->AddMessage(message, field), object);
343  } else {
344  return parse(reflection->MutableMessage(message, field), object);
345  }
346  break;
347  default:
348  return Error("Not expecting a JSON object for field '" +
349  field->name() + "'");
350  }
351  return Nothing();
352  }
353 
354  Try<Nothing> operator()(const JSON::String& string) const
355  {
356  switch (field->type()) {
357  case google::protobuf::FieldDescriptor::TYPE_STRING:
358  if (field->is_repeated()) {
359  reflection->AddString(message, field, string.value);
360  } else {
361  reflection->SetString(message, field, string.value);
362  }
363  break;
364  case google::protobuf::FieldDescriptor::TYPE_BYTES: {
365  Try<std::string> decode = base64::decode(string.value);
366 
367  if (decode.isError()) {
368  return Error("Failed to base64 decode bytes field"
369  " '" + field->name() + "': " + decode.error());
370  }
371 
372  if (field->is_repeated()) {
373  reflection->AddString(message, field, decode.get());
374  } else {
375  reflection->SetString(message, field, decode.get());
376  }
377  break;
378  }
379  case google::protobuf::FieldDescriptor::TYPE_ENUM: {
380  const google::protobuf::EnumValueDescriptor* descriptor =
381  field->enum_type()->FindValueByName(string.value);
382 
383  if (descriptor == NULL) {
384  return Error("Failed to find enum for '" + string.value + "'");
385  }
386 
387  if (field->is_repeated()) {
388  reflection->AddEnum(message, field, descriptor);
389  } else {
390  reflection->SetEnum(message, field, descriptor);
391  }
392  break;
393  }
394  default:
395  return Error("Not expecting a JSON string for field '" +
396  field->name() + "'");
397  }
398  return Nothing();
399  }

535 // NOTE: This struct is used by the public parse<T>() function below. See

567 // Partial specialization for RepeatedPtrField<T> to parse a sequence of
568 // protobuf messages from a JSON::Array by repeatedly invoking Parse<T> to
569 // facilitate conversions like JSON::Array -> Resources.
570 // NOTE: This struct is used by the public parse<T>() function below. See
571 // comments there for the reason why we opted for this design.
572 template <typename T>
573 struct Parse<google::protobuf::RepeatedPtrField<T>>
574 {
575  Try<google::protobuf::RepeatedPtrField<T>> operator()(
576  const JSON::Value& value)
577  {
578  static_assert(std::is_convertible<T*, google::protobuf::Message*>::value,
579  "T must be a protobuf message");
580 
581  const JSON::Array* array = boost::get<JSON::Array>(&value);
582  if (array == NULL) {
583  return Error("Expecting a JSON array");
584  }
585 
586  google::protobuf::RepeatedPtrField<T> collection;
587  collection.Reserve(static_cast<int>(array->values.size()));
588 
589  // Parse messages one by one and propagate an error if it happens.
590  foreach (const JSON::Value& elem, array->values) {
591  Try<T> message = Parse<T>()(elem);
592  if (message.isError()) {
593  return Error(message.error());
594  }
595 
596  collection.Add()->CopyFrom(message.get());
597  }
598 
599  return collection;
600  }

623 // `json` function for protobuf messages. Refer to `jsonify.hpp` for details.
624 // TODO(mpark): This currently uses the default value for optional fields with
625 // a default that are not set, but we may want to revisit this decision.
626 inline void json(ObjectWriter* writer, const google::protobuf::Message& message)
627 {
628  using google::protobuf::FieldDescriptor;
629 
630  const google::protobuf::Descriptor* descriptor = message.GetDescriptor();
631  const google::protobuf::Reflection* reflection = message.GetReflection();
632 
633  // We first look through all the possible fields to determine both the set
634  // fields __and__ the optional fields with a default that are not set.
635  // `Reflection::ListFields()` alone will only include set fields and
636  // is therefore insufficient.
637  int fieldCount = descriptor->field_count();
638  std::vector<const FieldDescriptor*> fields;
639  fields.reserve(fieldCount);
640  for (int i = 0; i < fieldCount; ++i) {
641  const FieldDescriptor* field = descriptor->field(i);
642  if (field->is_repeated()) {
643  if (reflection->FieldSize(message, field) > 0) {
644  // Has repeated field with members, output as JSON.
645  fields.push_back(field);
646  }
647  } else if (reflection->HasField(message, field) ||
648  field->has_default_value()) {
649  // Field is set or has default, output as JSON.
650  fields.push_back(field);
651  }
652  }
653 
654  foreach (const FieldDescriptor* field, fields) {
655  if (field->is_repeated()) {
656  writer->field(
657  field->name(),
658  [&field, &reflection, &message](JSON::ArrayWriter* writer) {
659  int fieldSize = reflection->FieldSize(message, field);
660  for (int i = 0; i < fieldSize; ++i) {
661  switch (field->cpp_type()) {
662  case FieldDescriptor::CPPTYPE_BOOL:
663  writer->element(
664  reflection->GetRepeatedBool(message, field, i));
665  break;
666  case FieldDescriptor::CPPTYPE_INT32:
667  writer->element(
668  reflection->GetRepeatedInt32(message, field, i));
669  break;
670  case FieldDescriptor::CPPTYPE_INT64:
671  writer->element(
672  reflection->GetRepeatedInt64(message, field, i));
673  break;
674  case FieldDescriptor::CPPTYPE_UINT32:
675  writer->element(
676  reflection->GetRepeatedUInt32(message, field, i));
677  break;
678  case FieldDescriptor::CPPTYPE_UINT64:
679  writer->element(
680  reflection->GetRepeatedUInt64(message, field, i));
681  break;
682  case FieldDescriptor::CPPTYPE_FLOAT:
683  writer->element(
684  reflection->GetRepeatedFloat(message, field, i));
685  break;
686  case FieldDescriptor::CPPTYPE_DOUBLE:
687  writer->element(
688  reflection->GetRepeatedDouble(message, field, i));
689  break;
690  case FieldDescriptor::CPPTYPE_MESSAGE:
691  writer->element(
692  reflection->GetRepeatedMessage(message, field, i));
693  break;
694  case FieldDescriptor::CPPTYPE_ENUM:
695  writer->element(
696  reflection->GetRepeatedEnum(message, field, i)->name());
697  break;
698  case FieldDescriptor::CPPTYPE_STRING:
699  const std::string& s = reflection->GetRepeatedStringReference(
700  message, field, i, NULL);
701  if (field->type() == FieldDescriptor::TYPE_BYTES) {
702  writer->element(base64::encode(s));
703  } else {
704  writer->element(s);
705  }
706  break;
707  }
708  }
709  });
710  } else {
711  switch (field->cpp_type()) {
712  case FieldDescriptor::CPPTYPE_BOOL:
713  writer->field(field->name(), reflection->GetBool(message, field));
714  break;

772  if (field->is_repeated()) {

774  // Has repeated field with members, output as JSON.
775  fields.push_back(field);
776  }
