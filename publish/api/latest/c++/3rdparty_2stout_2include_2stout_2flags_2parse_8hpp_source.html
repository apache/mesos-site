87 template <>
88 inline Try<JSON::Object> parse(const std::string& value)
89 {
90  // A value that already starts with 'file://' will properly be
91  // loaded from the file and put into 'value' but if it starts with
92  // '/' we need to explicitly handle it for backwards compatibility
93  // reasons (because we used to handle it before we introduced the
94  // 'fetch' mechanism for flags that first fetches the data from URIs
95  // such as 'file://').
96  if (strings::startsWith(value, "/")) {
97  LOG(WARNING) << "Specifying an absolute filename to read a command line "
98  "option out of without using 'file:// is deprecated and "
99  "will be removed in a future release. Simply adding "
100  "'file://' to the beginning of the path should eliminate "
101  "this warning.";
102 
103  Try<std::string> read = os::read(value);
104  if (read.isError()) {
105  return Error("Error reading file '" + value + "': " + read.error());
106  }
107  return JSON::parse<JSON::Object>(read.get());
108  }
109  return JSON::parse<JSON::Object>(value);
110 }
