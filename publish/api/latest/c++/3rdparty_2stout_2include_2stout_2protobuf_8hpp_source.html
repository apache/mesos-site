61 // Write out the given protobuf to the specified file descriptor by
62 // first writing out the length of the protobuf followed by the
63 // contents.
64 // NOTE: On error, this may have written partial data to the file.
65 inline Try<Nothing> write(int_fd fd, const google::protobuf::Message& message)
66 {
67  if (!message.IsInitialized()) {
68  return Error(message.InitializationErrorString() +
69  " is required but not initialized");
70  }
71 
72  // First write the size of the protobuf.
73  uint32_t size = message.ByteSize();
74  std::string bytes((char*) &size, sizeof(size));
75 
76  Try<Nothing> result = os::write(fd, bytes);
77  if (result.isError()) {
78  return Error("Failed to write size: " + result.error());
79  }
80 
81 #ifdef __WINDOWS__
82  if (!message.SerializeToFileDescriptor(fd.crt())) {
83 #else
84  if (!message.SerializeToFileDescriptor(fd)) {
85 #endif
86  return Error("Failed to write/serialize message");
87  }
88 
89  return Nothing();
90 }

92 // Write out the given sequence of protobuf messages to the
93 // specified file descriptor by repeatedly invoking write
94 // on each of the messages.
95 // NOTE: On error, this may have written partial data to the file.
96 template <typename T>
97 Try<Nothing> write(
98  int_fd fd, const google::protobuf::RepeatedPtrField<T>& messages)
99 {
100  foreach (const T& message, messages) {
101  Try<Nothing> result = write(fd, message);
102  if (result.isError()) {
103  return Error(result.error());
104  }
105  }
106 
107  return Nothing();
108 }

172 template <typename T>
173 Try<T> deserialize(const std::string& value)
174 {
175  T t;
176  (void) static_cast<google::protobuf::Message*>(&t);
177 
178  // Verify that the size of `value` fits into `ArrayInputStream`'s
179  // constructor. The maximum size of a proto2 message is 64 MB, so it is
180  // unlikely that we will hit this limit, but since an arbitrary string can be
181  // passed in, we include this check to be safe.
182  CHECK_LE(value.size(), static_cast<size_t>(std::numeric_limits<int>::max()));
183  google::protobuf::io::ArrayInputStream stream(
184  value.data(),
185  static_cast<int>(value.size()));
186  if (!t.ParseFromZeroCopyStream(&stream)) {
187  return Error("Failed to deserialize " + t.GetDescriptor()->full_name());
188  }
189  return t;
190 }

208 // Reads a single message of type T from the file by first reading the
209 // "size" followed by the contents (as written by 'write' above).
210 // NOTE: This struct is used by the public 'read' function.
211 // See comments there for the reason why we need this.
212 template <typename T>
213 struct Read
214 {
215  Result<T> operator()(int_fd fd, bool ignorePartial, bool undoFailed)
216  {
217  off_t offset = 0;
218 
219  if (undoFailed) {
220  // Save the offset so we can re-adjust if something goes wrong.
221  Try<off_t> lseek = os::lseek(fd, offset, SEEK_CUR);
222  if (lseek.isError()) {
223  return Error(lseek.error());
224  }
225 
226  offset = lseek.get();
227  }
228 
229  uint32_t size;
230  Result<std::string> result = os::read(fd, sizeof(size));
231 
232  if (result.isError()) {
233  if (undoFailed) {
234  os::lseek(fd, offset, SEEK_SET);
235  }
236  return Error("Failed to read size: " + result.error());
237  } else if (result.isNone()) {
238  return None(); // No more protobufs to read.
239  } else if (result.get().size() < sizeof(size)) {
240  // Hit EOF unexpectedly.
241  if (undoFailed) {
242  // Restore the offset to before the size read.
243  os::lseek(fd, offset, SEEK_SET);
244  }
245  if (ignorePartial) {
246  return None();
247  }
248  return Error(
249  "Failed to read size: hit EOF unexpectedly, possible corruption");
250  }
251 
252  // Parse the size from the bytes.
253  memcpy((void*) &size, (void*) result.get().data(), sizeof(size));
254 
255  // NOTE: Instead of specifically checking for corruption in 'size',
256  // we simply try to read 'size' bytes. If we hit EOF early, it is an
257  // indication of corruption.
258  result = os::read(fd, size);
259 
260  if (result.isError()) {
261  if (undoFailed) {
262  // Restore the offset to before the size read.
263  os::lseek(fd, offset, SEEK_SET);
264  }
265  return Error("Failed to read message: " + result.error());
266  } else if (result.isNone() || result.get().size() < size) {
267  // Hit EOF unexpectedly.
268  if (undoFailed) {
269  // Restore the offset to before the size read.
270  os::lseek(fd, offset, SEEK_SET);
271  }
272  if (ignorePartial) {
273  return None();
274  }
275  return Error("Failed to read message of size " + stringify(size) +
276  " bytes: hit EOF unexpectedly, possible corruption");
277  }
278 
279  // Parse the protobuf from the string.
280  // NOTE: We need to capture a const reference to the data because it
281  // must outlive the creation of ArrayInputStream.
282  const std::string& data = result.get();
283 
284  // Verify that the size of `data` fits into `ArrayInputStream`'s
285  // constructor. The maximum size of a proto2 message is 64 MB, so it is
286  // unlikely that we will hit this limit, but since an arbitrary string can
287  // be passed in, we include this check to be safe.
288  CHECK_LE(data.size(), static_cast<size_t>(std::numeric_limits<int>::max()));
289  T message;
290  google::protobuf::io::ArrayInputStream stream(
291  data.data(),
292  static_cast<int>(data.size()));
293 
294  if (!message.ParseFromZeroCopyStream(&stream)) {
295  if (undoFailed) {
296  // Restore the offset to before the size read.
297  os::lseek(fd, offset, SEEK_SET);
298  }
299  return Error("Failed to deserialize message");
300  }
301 
302  return message;
303  }
304 };

307 // Partial specialization for RepeatedPtrField<T> to read a sequence
308 // of protobuf messages from a given fd by repeatedly invoking
309 // Read<T> until None is reached, which we treat as EOF.
310 // NOTE: This struct is used by the public 'read' function.
311 // See comments there for the reason why we need this.
312 template <typename T>
313 struct Read<google::protobuf::RepeatedPtrField<T>>
314 {
315  Result<google::protobuf::RepeatedPtrField<T>> operator()(
316  int_fd fd, bool ignorePartial, bool undoFailed)
317  {
318  google::protobuf::RepeatedPtrField<T> result;
319  for (;;) {
320  Result<T> message = Read<T>()(fd, ignorePartial, undoFailed);
321  if (message.isError()) {
322  return Error(message.error());
323  } else if (message.isNone()) {
324  break;
325  } else {
326  result.Add()->CopyFrom(message.get());
327  }
328  }
329  return result;
330  }
331 };

336 // Reads the protobuf message(s) from a given fd based on the format
337 // written by write() above. We use partial specialization of
338 // - internal::Read<T> vs
339 // - internal::Read<google::protobuf::RepeatedPtrField<T>>
340 // in order to determine whether T is a single protobuf message or
341 // a sequence of messages.
342 // If 'ignorePartial' is true, None() is returned when we unexpectedly
343 // hit EOF while reading the protobuf (e.g., partial write).
344 // If 'undoFailed' is true, failed read attempts will restore the file
345 // read/write file offset towards the initial callup position.
346 template <typename T>
347 Result<T> read(int_fd fd, bool ignorePartial = false, bool undoFailed = false)
348 {
349  return internal::Read<T>()(fd, ignorePartial, undoFailed);
350 }

393 struct Parser : boost::static_visitor<Try<Nothing>>
394 {
395  Parser(google::protobuf::Message* _message,
396  const google::protobuf::FieldDescriptor* _field)
397  : message(_message),
398  reflection(message->GetReflection()),
399  field(_field) {}
400 
401  Try<Nothing> operator()(const JSON::Object& object) const
402  {
403  switch (field->type()) {
404  case google::protobuf::FieldDescriptor::TYPE_MESSAGE:
405  // TODO(gilbert): We currently push up the nested error
406  // messages without wrapping the error message (due to
407  // the recursive nature of parse). We should pass along
408  // variable information in order to construct a helpful
409  // error message, e.g. "Failed to parse field 'a.b.c': ...".
410  if (field->is_repeated()) {
411  return parse(reflection->AddMessage(message, field), object);
412  } else {
413  return parse(reflection->MutableMessage(message, field), object);
414  }
415  break;
416  default:
417  return Error("Not expecting a JSON object for field '" +
418  field->name() + "'");
419  }
420  return Nothing();
421  }
422 
423  Try<Nothing> operator()(const JSON::String& string) const
424  {
425  switch (field->type()) {
426  case google::protobuf::FieldDescriptor::TYPE_STRING:
427  if (field->is_repeated()) {
428  reflection->AddString(message, field, string.value);
429  } else {
430  reflection->SetString(message, field, string.value);
431  }
432  break;
433  case google::protobuf::FieldDescriptor::TYPE_BYTES: {
434  Try<std::string> decode = base64::decode(string.value);
435 
436  if (decode.isError()) {
437  return Error("Failed to base64 decode bytes field"
438  " '" + field->name() + "': " + decode.error());
439  }
440 
441  if (field->is_repeated()) {
442  reflection->AddString(message, field, decode.get());
443  } else {
444  reflection->SetString(message, field, decode.get());
445  }
446  break;
447  }
448  case google::protobuf::FieldDescriptor::TYPE_ENUM: {
449  const google::protobuf::EnumValueDescriptor* descriptor =
450  field->enum_type()->FindValueByName(string.value);
451 
452  if (descriptor == nullptr) {
453  return Error("Failed to find enum for '" + string.value + "'");
454  }
455 
456  if (field->is_repeated()) {
457  reflection->AddEnum(message, field, descriptor);
458  } else {
459  reflection->SetEnum(message, field, descriptor);
460  }
461  break;
462  }
463  default:
464  return Error("Not expecting a JSON string for field '" +
465  field->name() + "'");
466  }
467  return Nothing();
468  }

603 // Parses a single protobuf message of type T from a JSON::Object.
604 // NOTE: This struct is used by the public parse<T>() function below. See
605 // comments there for the reason why we opted for this design.
606 template <typename T>
607 struct Parse
608 {
609  Try<T> operator()(const JSON::Value& value)
610  {
611  static_assert(std::is_convertible<T*, google::protobuf::Message*>::value,
612  "T must be a protobuf message");
613 
614  const JSON::Object* object = boost::get<JSON::Object>(&value);
615  if (object == nullptr) {
616  return Error("Expecting a JSON object");
617  }
618 
619  T message;
620 
621  Try<Nothing> parse = internal::parse(&message, *object);
622  if (parse.isError()) {
623  return Error(parse.error());
624  }
625 
626  if (!message.IsInitialized()) {
627  return Error("Missing required fields: " +
628  message.InitializationErrorString());
629  }
630 
631  return message;
632  }

636 // Partial specialization for RepeatedPtrField<T> to parse a sequence of
637 // protobuf messages from a JSON::Array by repeatedly invoking Parse<T> to
638 // facilitate conversions like JSON::Array -> Resources.
639 // NOTE: This struct is used by the public parse<T>() function below. See
640 // comments there for the reason why we opted for this design.
641 template <typename T>
642 struct Parse<google::protobuf::RepeatedPtrField<T>>
643 {
644  Try<google::protobuf::RepeatedPtrField<T>> operator()(
645  const JSON::Value& value)
646  {
647  static_assert(std::is_convertible<T*, google::protobuf::Message*>::value,
648  "T must be a protobuf message");
649 
650  const JSON::Array* array = boost::get<JSON::Array>(&value);
651  if (array == nullptr) {
652  return Error("Expecting a JSON array");
653  }
654 
655  google::protobuf::RepeatedPtrField<T> collection;
656  collection.Reserve(static_cast<int>(array->values.size()));
657 
658  // Parse messages one by one and propagate an error if it happens.
659  foreach (const JSON::Value& elem, array->values) {
660  Try<T> message = Parse<T>()(elem);
661  if (message.isError()) {
662  return Error(message.error());
663  }
664 
665  collection.Add()->CopyFrom(message.get());
666  }
667 
668  return collection;
669  }
