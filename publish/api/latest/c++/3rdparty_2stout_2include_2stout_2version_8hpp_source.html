26 // This class provides convenience routines for version checks.
27 //
28 // Ideally, the components would be called simply major, minor and
29 // patch. However, GNU libstdc++ already defines these as macros for
30 // compatibility reasons (man 3 makedev for more information) implicitly
31 // included in every compilation.
32 //
33 // TODO(karya): Consider adding support for more than 3 components, and
34 // compatibility operators.
35 // TODO(karya): Add support for labels and build metadata. Consider
36 // semantic versioning (http://semvar.org/) for specs.
37 struct Version
38 {
39  // Expect the string in the following format:
40  // <major>[.<minor>[.<patch>]]
41  // Missing components are treated as zero.
42  static Try<Version> parse(const std::string& s)
43  {
44  const size_t maxComponents = 3;
45 
46  // Use only the part before '-', i.e. strip and discard the tags
47  // and labels.
48  // TODO(karya): Once we have support for labels and tags, we
49  // should not discard the remaining string.
50  std::vector<std::string> split =
51  strings::split(strings::split(s, "-")[0], ".");
52 
53  if (split.size() > maxComponents) {
54  return Error("Version string has " + stringify(split.size()) +
55  " components; maximum " + stringify(maxComponents) +
56  " components allowed");
57  }
58 
59  int components[maxComponents] = {0};
60 
61  for (size_t i = 0; i < split.size(); i++) {
62  Try<int> result = numify<int>(split[i]);
63  if (result.isError()) {
64  return Error("Invalid version component '" + split[i] + "': " +
65  result.error());
66  }
67  components[i] = result.get();
68  }
69 
70  return Version(components[0], components[1], components[2]);
71  }
