32  static Try<Bytes> parse(const std::string& s)
33  {
34  size_t index = 0;
35 
36  while (index < s.size()) {
37  if (isdigit(s[index])) {
38  index++;
39  continue;
40  } else if (s[index] == '.') {
41  return Error("Fractional bytes '" + s + "'");
42  }
43 
44  Try<uint64_t> value = numify<uint64_t>(s.substr(0, index));
45 
46  if (value.isError()) {
47  return Error(value.error());
48  }
49 
50  const std::string& unit = strings::upper(s.substr(index));
51 
52  if (unit == "B") {
53  return Bytes(value.get(), BYTES);
54  } else if (unit == "KB") {
55  return Bytes(value.get(), KILOBYTES);
56  } else if (unit == "MB") {
57  return Bytes(value.get(), MEGABYTES);
58  } else if (unit == "GB") {
59  return Bytes(value.get(), GIGABYTES);
60  } else if (unit == "TB") {
61  return Bytes(value.get(), TERABYTES);
62  } else {
63  return Error("Unknown bytes unit '" + unit + "'");
64  }
65  }
66  return Error("Invalid bytes '" + s + "'");
67  }
