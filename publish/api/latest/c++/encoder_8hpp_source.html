114  static std::string encode(Message* message)
115  {
116  std::ostringstream out;
117 
118  if (message != nullptr) {
119  out << "POST ";
120  // Nothing keeps the 'id' component of a PID from being an empty
121  // string which would create a malformed path that has two
122  // '//' unless we check for it explicitly.
123  // TODO(benh): Make the 'id' part of a PID optional so when it's
124  // missing it's clear that we're simply addressing an ip:port.
125  if (message->to.id != "") {
126  out << "/" << message->to.id;
127  }
128 
129  out << "/" << message->name << " HTTP/1.1\r\n"
130  << "User-Agent: libprocess/" << message->from << "\r\n"
131  << "Libprocess-From: " << message->from << "\r\n"
132  << "Connection: Keep-Alive\r\n"
133  << "Host: \r\n";
134 
135  if (message->body.size() > 0) {
136  out << "Transfer-Encoding: chunked\r\n\r\n"
137  << std::hex << message->body.size() << "\r\n";
138  out.write(message->body.data(), message->body.size());
139  out << "\r\n"
140  << "0\r\n"
141  << "\r\n";
142  } else {
143  out << "\r\n";
144  }
145  }
146 
147  return out.str();
148  }
149 
150 private:
151  Message* message;
152 };

163  static std::string encode(
164  const http::Response& response,
165  const http::Request& request)
166  {
167  std::ostringstream out;
168 
169  // TODO(benh): Check version?
170 
171  out << "HTTP/1.1 " << response.status << "\r\n";
172 
173  auto headers = response.headers;
174 
175  // HTTP 1.1 requires the "Date" header. In the future once we
176  // start checking the version (above) then we can conditionally
177  // add this header, but for now, we always do.
178  time_t rawtime;
179  time(&rawtime);
180 
181  char date[256];
182 
183  tm tm_;
184  PCHECK(os::gmtime_r(&rawtime, &tm_) != nullptr)
185  << "Failed to convert the current time to a tm struct "
186  << "using os::gmtime_r()";
187 
188  // TODO(benh): Check return code of strftime!
189  strftime(date, 256, "%a, %d %b %Y %H:%M:%S GMT", &tm_);
190 
191  headers["Date"] = date;
192 
193  // Should we compress this response?
194  std::string body = response.body;
195 
196  if (response.type == http::Response::BODY &&
197  response.body.length() >= GZIP_MINIMUM_BODY_LENGTH &&
198  !headers.contains("Content-Encoding") &&
199  request.acceptsEncoding("gzip")) {
200  Try<std::string> compressed = gzip::compress(body);
201  if (compressed.isError()) {
202  LOG(WARNING) << "Failed to gzip response body: " << compressed.error();
203  } else {
204  body = compressed.get();
205  headers["Content-Length"] = stringify(body.length());
206  headers["Content-Encoding"] = "gzip";
207  }
208  }
209 
210  foreachpair (const std::string& key, const std::string& value, headers) {
211  out << key << ": " << value << "\r\n";
212  }
213 
214  // Add a Content-Length header if the response is of type "none"
215  // or "body" and no Content-Length header has been supplied.
216  if (response.type == http::Response::NONE &&
217  !headers.contains("Content-Length")) {
218  out << "Content-Length: 0\r\n";
219  } else if (response.type == http::Response::BODY &&
220  !headers.contains("Content-Length")) {
221  out << "Content-Length: " << body.size() << "\r\n";
222  }
223 
224  // Use a CRLF to mark end of headers.
225  out << "\r\n";
226 
227  // Add the body if necessary.
228  if (response.type == http::Response::BODY) {
229  // If the Content-Length header was supplied, only write as much data
230  // as the length specifies.
231  Result<uint32_t> length = numify<uint32_t>(headers.get("Content-Length"));
232  if (length.isSome() && length.get() <= body.length()) {
233  out.write(body.data(), length.get());
234  } else {
235  out.write(body.data(), body.size());
236  }
237  }
238 
239  return out.str();
240  }
241 };
