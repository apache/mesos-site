40 // comparing std::type_info of the member function pointers. Because
41 // of this, the method function pointer passed to either
42 // FUTURE_DISPATCH or DROP_DISPATCH must match exactly the member

45 // function and that a derived class overrides, and if in unit tests

144 // This action invokes an "inner" action but captures the result and
145 // stores a copy of it in a future. Note that this is implemented
146 // similarly to the IgnoreResult action, which relies on the cast
147 // operator to Action<F> which must occur (implicitly) before the
148 // expression has completed, hence we can pass the Future<R>* all the
149 // way through to our action "implementation".
150 template <typename R, typename A>
151 class FutureResultAction
152 {
153 public:
154  explicit FutureResultAction(process::Future<R>* future, const A& action)
155  : future(future),
156  action(action) {}
157 
158  template <typename F>
159  operator ::testing::Action<F>() const
160  {
161  return ::testing::Action<F>(new Implementation<F>(future, action));
162  }
163 
164 private:
165  template <typename F>
166  class Implementation : public ::testing::ActionInterface<F>
167  {
168  public:
169  explicit Implementation(process::Future<R>* future, const A& action)
170  : action(action)
171  {
172  *future = promise.future();
173  }
174 
175  virtual typename ::testing::ActionInterface<F>::Result Perform(
176  const typename ::testing::ActionInterface<F>::ArgumentTuple& args)
177  {
178  const typename ::testing::ActionInterface<F>::Result result =
179  action.Perform(args);
180  promise.set(result);
181  return result;
182  }
183 
184  private:
185  // Not copyable, not assignable.
186  Implementation(const Implementation&);
187  Implementation& operator=(const Implementation&);
188 
189  process::Promise<R> promise;
190  const ::testing::Action<F> action;
191  };
192 
193  process::Future<R>* future;
194  const A action;
195 };
196

231 // A definition of a libprocess filter to enable waiting for events
232 // (such as messages or dispatches) via in tests. This is not meant to
233 // be used directly by tests; tests should use macros like
234 // FUTURE_MESSAGE and FUTURE_DISPATCH instead.
235 class TestsFilter : public Filter
236 {
237 public:
238  TestsFilter() = default;
239 
240  virtual bool filter(const MessageEvent& event) { return handle(event); }
241  virtual bool filter(const DispatchEvent& event) { return handle(event); }
242  virtual bool filter(const HttpEvent& event) { return handle(event); }
243  virtual bool filter(const ExitedEvent& event) { return handle(event); }
244 
245  template <typename T>
246  bool handle(const T& t)
247  {
248  synchronized (mutex) {
249  return mock.filter(t);
250  }
251  }
