32 // Returns a gzip compressed version of the provided string.
33 // The compression level should be within the range [-1, 9].
34 // See zlib.h:
35 // #define Z_NO_COMPRESSION 0
36 // #define Z_BEST_SPEED 1
37 // #define Z_BEST_COMPRESSION 9
38 // #define Z_DEFAULT_COMPRESSION (-1)
39 inline Try<std::string> compress(
40  const std::string& decompressed,
41  int level = Z_DEFAULT_COMPRESSION)
42 {
43  // Verify the level is within range.
44  if (!(level == Z_DEFAULT_COMPRESSION ||
45  (level >= Z_NO_COMPRESSION && level <= Z_BEST_COMPRESSION))) {
46  return Error("Invalid compression level: " + stringify(level));
47  }
48 
49  z_stream_s stream;
50  stream.next_in =
51  const_cast<Bytef*>(reinterpret_cast<const Bytef*>(decompressed.data()));
52  stream.avail_in = decompressed.length();
53  stream.zalloc = Z_NULL;
54  stream.zfree = Z_NULL;
55  stream.opaque = Z_NULL;
56 
57  int code = deflateInit2(
58  &stream,
59  level, // Compression level.
60  Z_DEFLATED, // Compression method.
61  MAX_WBITS + 16, // Zlib magic for gzip compression / decompression.
62  8, // Default memLevel value.
63  Z_DEFAULT_STRATEGY);
64 
65  if (code != Z_OK) {
66  return Error("Failed to initialize zlib: " + std::string(stream.msg));
67  }
68 
69  // Build up the compressed result.
70  Bytef buffer[GZIP_BUFFER_SIZE];
71  std::string result = "";
72  do {
73  stream.next_out = buffer;
74  stream.avail_out = GZIP_BUFFER_SIZE;
75  code = deflate(&stream, stream.avail_in > 0 ? Z_NO_FLUSH : Z_FINISH);
76 
77  if (code != Z_OK && code != Z_STREAM_END) {
78  Error error(std::string(stream.msg));
79  deflateEnd(&stream);
80  return error;
81  }
82 
83  // Consume output and reset the buffer.
84  result.append(
85  reinterpret_cast<char*>(buffer),
86  GZIP_BUFFER_SIZE - stream.avail_out);
87  stream.next_out = buffer;
88  stream.avail_out = GZIP_BUFFER_SIZE;
89  } while (code != Z_STREAM_END);
90 
91  code = deflateEnd(&stream);
92  if (code != Z_OK) {
93  return Error("Failed to clean up zlib: " + std::string(stream.msg));
94  }
95  return result;
96 }
