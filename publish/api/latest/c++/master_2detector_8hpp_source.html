46 // An abstraction of a Master detector which can be used to
47 // detect the leading master from a group.
48 class MasterDetector
49 {
50 public:
51  // Attempts to create a master detector for the specified mechanism.
52  // The mechanism should be one of:
53  // - host:port
54  // - zk://host1:port1,host2:port2,.../path
55  // - zk://username:password@host1:port1,host2:port2,.../path
56  static Try<MasterDetector*> create(const Option<std::string>& _mechanism);
57  virtual ~MasterDetector() = 0;
58 
59  // Returns MasterInfo after an election has occurred and the elected
60  // master is different than that specified (if any), or NONE if an
61  // election occurs and no master is elected (e.g., all masters are
62  // lost). A failed future is returned if the detector is unable to
63  // detect the leading master due to a non-retryable error.
64  // Note that the detector transparently tries to recover from
65  // retryable errors.
66  // The future is never discarded unless it stays pending when the
67  // detector destructs.
68  //
69  // The 'previous' result (if any) should be passed back if this
70  // method is called repeatedly so the detector only returns when it
71  // gets a different result.
72  virtual process::Future<Option<MasterInfo> > detect(
73  const Option<MasterInfo>& previous = None()) = 0;
74 };
