162  void addTask(Task* task)
163  {
164  const TaskID& taskId = task->task_id();
165  const FrameworkID& frameworkId = task->framework_id();
166 
167  CHECK(!tasks[frameworkId].contains(taskId))
168  << "Duplicate task " << taskId << " of framework " << frameworkId;
169 
170  tasks[frameworkId][taskId] = task;
171 
172  if (!protobuf::isTerminalState(task->state())) {
173  usedResources[frameworkId] += task->resources();
174  }
175 
176  LOG(INFO) << "Adding task " << taskId
177  << " with resources " << task->resources()
178  << " on slave " << id << " (" << info.hostname() << ")";

181  // Notification of task termination, for resource accounting.
182  // TODO(bmahler): This is a hack for performance. We need to
183  // maintain resource counters because computing task resources
184  // functionally for all tasks is expensive, for now.
185  void taskTerminated(Task* task)
186  {
187  const TaskID& taskId = task->task_id();
188  const FrameworkID& frameworkId = task->framework_id();
189 
190  CHECK(protobuf::isTerminalState(task->state()));
191  CHECK(tasks[frameworkId].contains(taskId))
192  << "Unknown task " << taskId << " of framework " << frameworkId;
193 
194  usedResources[frameworkId] -= task->resources();
195  if (!tasks.contains(frameworkId) && !executors.contains(frameworkId)) {
196  usedResources.erase(frameworkId);
197  }
198  }

200  void removeTask(Task* task)
201  {
202  const TaskID& taskId = task->task_id();
203  const FrameworkID& frameworkId = task->framework_id();
204 
205  CHECK(tasks[frameworkId].contains(taskId))
206  << "Unknown task " << taskId << " of framework " << frameworkId;
207 
208  if (!protobuf::isTerminalState(task->state())) {
209  usedResources[frameworkId] -= task->resources();
210  if (!tasks.contains(frameworkId) && !executors.contains(frameworkId)) {
211  usedResources.erase(frameworkId);
212  }
213  }
214 
215  tasks[frameworkId].erase(taskId);
216  if (tasks[frameworkId].empty()) {
217  tasks.erase(frameworkId);
218  }
219 
220  killedTasks.remove(frameworkId, taskId);
221  }

955  private:
956  // Heuristically tries to determine whether a quota request could
957  // reasonably be satisfied given the current cluster capacity. The
958  // goal is to determine whether a user may accidentally request an
959  // amount of resources that would prevent frameworks without quota
960  // from getting any offers. A force flag will allow users to bypass
961  // this check.
962  //
963  // The heuristic tests whether the total quota, including the new
964  // request, does not exceed the sum of non-static cluster resources,
965  // i.e. the following inequality holds:
966  // total - statically reserved >= total quota + quota request
967  //
968  // Please be advised that:
969  // * It is up to an allocator how to satisfy quota (for example,
970  // what resources to account towards quota, as well as which
971  // resources to consider allocatable for quota).
972  // * Even if there are enough resources at the moment of this check,
973  // agents may terminate at any time, rendering the cluster under
974  // quota.
975  Option<Error> capacityHeuristic(
976  const mesos::quota::QuotaInfo& request) const;
977 
978  // We always want to rescind offers after the capacity heuristic. The
979  // reason for this is the race between the allocator and the master:
980  // it can happen that there are not enough free resources at the
981  // allocator's disposal when it is notified about the quota request,
982  // but at this point it's too late to rescind.
983  //
984  // While rescinding, we adhere to the following rules:
985  // * Rescind at least as many resources as there are in the quota request.
986  // * Rescind all offers from an agent in order to make the potential
987  // offer bigger, which increases the chances that a quota'ed framework
988  // will be able to use the offer.
989  // * Rescind offers from at least `numF` agents to make it possible
990  // (but not guaranteed, due to fair sharing) that each framework in
991  // the role for which quota is set gets an offer (`numF` is the
992  // number of frameworks in the quota'ed role). Though this is not
993  // strictly necessary, we think this will increase the debugability
994  // and will improve user experience.
995  //
996  // TODO(alexr): Consider removing this function once offer management
997  // (including rescinding) is moved to allocator.
