113  inline std::string basename() const
114  {
115  if (value.empty()) {
116  return std::string(".");
117  }
118 
119  size_t end = value.size() - 1;
120 
121  // Remove trailing slashes.
122  if (value[end] == os::PATH_SEPARATOR) {
123  end = value.find_last_not_of(os::PATH_SEPARATOR, end);
124 
125  // Paths containing only slashes result into "/".
126  if (end == std::string::npos) {
127  return stringify(os::PATH_SEPARATOR);
128  }
129  }
130 
131  // 'start' should point towards the character after the last slash
132  // that is non trailing.
133  size_t start = value.find_last_of(os::PATH_SEPARATOR, end);
134 
135  if (start == std::string::npos) {
136  start = 0;
137  } else {
138  start++;
139  }
140 
141  return value.substr(start, end + 1 - start);
142  }

179 
180  // Remove trailing slashes.
181  if (value[end] == os::PATH_SEPARATOR) {
182  end = value.find_last_not_of(os::PATH_SEPARATOR, end);
183  }
184 
185  // Remove anything trailing the last slash.
186  end = value.find_last_of(os::PATH_SEPARATOR, end);
187 
188  // Paths containing no slashes result in ".".
189  if (end == std::string::npos) {
190  return std::string(".");

192 
193  // Paths containing only slashes result in "/".
194  if (end == 0) {
195  return stringify(os::PATH_SEPARATOR);
196  }
197 
198  // 'end' should point towards the last non slash character
199  // preceding the last slash.
200  end = value.find_last_not_of(os::PATH_SEPARATOR, end);
201 
202  // Paths containing no non slash characters result in "/".
203  if (end == std::string::npos) {
204  return stringify(os::PATH_SEPARATOR);
205  }
206
