20 // The Scope construct provides a means to ensure execution of
21 // statements upon leaving the current scope. Typically used for
22 // performing cleanup as an alternative to introducing a special
23 // class performing the cleanup via RAII.
24 //
25 // E.g.
26 // {
27 // FILE* f = fopen('/tmp/file');
28 //
29 // if (f == NULL) {
30 // return ErrnoError();
31 // }
32 //
33 // Scope scope;
34 // scope.at_exit([&]() { fclose(f) });
35 //
36 // char* buffer = new char[BUFSIZ];
37 // scope.at_exit([&]() { delete buffer });
38 //
39 // size_t read = ::fread(buffer, 1, BUFSIZ, file);
40 //
41 // if (ferror(file)) {
42 // return ErrnoError();
43 // }
44 //
45 // return string(buffer, size);
46 // }
47 class Scope
48 {
49 public:
50  ~Scope()
51  {
52  // The `at_exits` run in the reverse order that they were added.
53  for (auto it = at_exits.rbegin(); it != at_exits.rend(); ++it) {
54  (*it)();
55  }
56  }
57 
58  template <typename F>
59  void at_exit(F&& f)
60  {
61  at_exits.push_back(f);
62  }
63 
64 private:
65  std::vector<lambda::function<void()>> at_exits;
66 };
