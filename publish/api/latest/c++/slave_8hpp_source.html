704  // Converts the message to an Event before sending.
705  template <typename Message>
706  bool send(const Message& message)
707  {
708  // We need to evolve the internal 'message' into a
709  // 'v1::executor::Event'.
710  return writer.write(encoder.encode(evolve(message)));
711  }
