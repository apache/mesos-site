106  process::Future<Result<T>> read()
107  {
108  if (!records.empty()) {
109  Result<T> record = std::move(records.front());
110  records.pop();
111  return record;
112  }
113 
114  if (error.isSome()) {
115  return process::Failure(error.get().message);
116  }
117 
118  if (done) {
119  return None();

123  new process::Promise<Result<T>>());
124  waiters.push(std::move(waiter));

167  void _consume(const process::Future<std::string>& read)
168  {
169  if (!read.isReady()) {
170  fail("Pipe::Reader failure: " +
171  (read.isFailed() ? read.failure() : "discarded"));
172  return;
173  }
174 
175  // Have we reached EOF?
176  if (read.get().empty()) {
177  complete();
178  return;
179  }
180 
181  Try<std::deque<Try<T>>> decode = decoder.decode(read.get());
182 
183  if (decode.isError()) {
184  fail("Decoder failure: " + decode.error());
185  return;
186  }
187 
188  foreach (const Try<T>& record, decode.get()) {
189  if (!waiters.empty()) {
190  waiters.front()->set(Result<T>(std::move(record)));
191  waiters.pop();
192  } else {
193  records.push(std::move(record));
194  }
195  }
196 
197  consume();
198  }
