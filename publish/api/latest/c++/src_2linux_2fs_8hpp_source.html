162 // Structure describing the per-process mounts as found in
163 // /proc/[pid]/mountinfo. In particular, entries in this table specify
164 // the propagation properties of mounts, information not present in
165 // the MountTable or /etc/fstab. Entry order is preserved when
166 // parsing /proc/[pid]/mountinfo.
167 struct MountInfoTable {
168  // Structure describing an individual /proc/[pid]/mountinfo entry.
169  // See the /proc/[pid]/mountinfo section in 'man proc' for further
170  // details on each field.
171  struct Entry {
172  static Try<Entry> parse(const std::string& s);
173 
174  Entry() {}
175 
176  // Returns the ID of the peer group in which this mount resides.
177  // Returns none if this mount is not a shared mount.
178  Option<int> shared() const;
179 
180  // Returns the ID of the peer group in which this mount's master
181  // resides in. Returns none if this mount is not a slave mount.
182  Option<int> master() const;
183 
184  int id; // mountinfo[1]: mount ID.
185  int parent; // mountinfo[2]: parent ID.
186  dev_t devno; // mountinfo[3]: st_dev.
187 
188  std::string root; // mountinfo[4]: root of the mount.
189  std::string target; // mountinfo[5]: mount point.
190 
191  // Filesystem independent (VFS) options, e.g., "rw,noatime".
192  std::string vfsOptions; // mountinfo[6]: per-mount options.
193  // Filesystem dependent options, e.g., "rw,memory" for a memory
194  // cgroup filesystem.
195  std::string fsOptions; // mountinfo[11]: per-block options.
196 
197  // Current possible optional fields include shared:X, master:X,
198  // propagate_from:X, unbindable.
199  std::string optionalFields; // mountinfo[7]: optional fields.
200 
201  // mountinfo[8] is a separator.
202 
203  std::string type; // mountinfo[9]: filesystem type.
204  std::string source; // mountinfo[10]: source dev, other.
205  };
