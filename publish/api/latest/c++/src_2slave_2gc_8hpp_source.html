43 // Provides an abstraction for removing files and directories after
44 // some point at which they are no longer considered necessary to keep
45 // around. The intent with this abstraction is to also easily enable
46 // implementations that may actually copy files and directories to
47 // "more" permanent storage (or provide any other hooks that might be
48 // useful, e.g., emailing users some time before their files are
49 // scheduled for removal).
50 class GarbageCollector
51 {
52 public:
53  GarbageCollector();
54  virtual ~GarbageCollector();
55 
56  // Schedules the specified path for removal after the specified
57  // duration of time has elapsed. If the path is already scheduled,
58  // this will reschedule the removal operation, and induce a discard
59  // on the previous future.
60  // The future will become ready when the path has been removed.
61  // The future will fail if the path did not exist, or on error.
62  // The future will be discarded if the path was unscheduled, or
63  // was rescheduled.
64  // Note that you currently cannot discard a returned future, instead
65  // you must call unschedule.
66  virtual process::Future<Nothing> schedule(
67  const Duration& d,
68  const std::string& path);
69 
70  // Unschedules the specified path for removal.
71  // The future will be true if the path has been unscheduled.
72  // The future will be false if the path is not scheduled for
73  // removal, or the path has already being removed.
74  // Note that you currently cannot discard a returned future.
75  virtual process::Future<bool> unschedule(const std::string& path);
76 
77  // Deletes all the directories, whose scheduled garbage collection time
78  // is within the next 'd' duration of time.
79  virtual void prune(const Duration& d);
80 
81 private:
82  GarbageCollectorProcess* process;
83 };
