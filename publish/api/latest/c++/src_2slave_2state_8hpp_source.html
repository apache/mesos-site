62 // This function performs recovery from the state stored at 'rootDir'.
63 // If the 'strict' flag is set, any errors encountered while
64 // recovering a state are considered fatal and hence the recovery is
65 // short-circuited and returns an error. There might be orphaned
66 // executors that need to be manually cleaned up. If the 'strict' flag
67 // is not set, any errors encountered are considered non-fatal and the
68 // recovery continues by recovering as much of the state as possible,
69 // while increasing the 'errors' count. Note that 'errors' on a struct
70 // includes the 'errors' encountered recursively. In other words,
71 // 'State.errors' is the sum total of all recovery errors.
72 Try<State> recover(const std::string& rootDir, bool strict);

113 // Thin wrapper to checkpoint data to disk and perform the necessary
114 // error checking. It checkpoints an instance of T at the given path.
115 // We can checkpoint anything as long as T is supported by
116 // internal::checkpoint. Currently the list of supported Ts are:
117 // - std::string
118 // - google::protobuf::Message
119 // - google::protobuf::RepeatedPtrField<T>
120 // - mesos::Resources
121 //
122 // NOTE: We provide atomic (all-or-nothing) semantics here by always
123 // writing to a temporary file first then using os::rename to atomically
124 // move it to the desired path.
125 template <typename T>
126 Try<Nothing> checkpoint(const std::string& path, const T& t)
127 {
128  // Create the base directory.
129  std::string base = Path(path).dirname();
130 
131  Try<Nothing> mkdir = os::mkdir(base);
132  if (mkdir.isError()) {
133  return Error("Failed to create directory '" + base + "': " + mkdir.error());
134  }
135 
136  // NOTE: We create the temporary file at 'base/XXXXXX' to make sure
137  // rename below does not cross devices (MESOS-2319).
138  //
139  // TODO(jieyu): It's possible that the temporary file becomes
140  // dangling if slave crashes or restarts while checkpointing.
141  // Consider adding a way to garbage collect them.
142  Try<std::string> temp = os::mktemp(path::join(base, "XXXXXX"));
143  if (temp.isError()) {
144  return Error("Failed to create temporary file: " + temp.error());
145  }
146 
147  // Now checkpoint the instance of T to the temporary file.
148  Try<Nothing> checkpoint = internal::checkpoint(temp.get(), t);
149  if (checkpoint.isError()) {
150  // Try removing the temporary file on error.
151  os::rm(temp.get());
152 
153  return Error("Failed to write temporary file '" + temp.get() +
154  "': " + checkpoint.error());
155  }
156 
157  // Rename the temporary file to the path.
158  Try<Nothing> rename = os::rename(temp.get(), path);
159  if (rename.isError()) {
160  // Try removing the temporary file on error.
161  os::rm(temp.get());
162 
163  return Error("Failed to rename '" + temp.get() + "' to '" +
164  path + "': " + rename.error());
165  }
166 
167  return Nothing();
168 }
