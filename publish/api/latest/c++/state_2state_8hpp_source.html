43 // An abstraction of "state" (possibly between multiple distributed
44 // components) represented by "variables" (effectively key/value
45 // pairs). Variables are versioned such that setting a variable in the
46 // state will only succeed if the variable has not changed since last
47 // fetched. Varying implementations of state provide varying
48 // replicated guarantees.
49 //
50 // Note that the semantics of 'fetch' and 'store' provide
51 // atomicity. That is, you can not store a variable that has changed
52 // since you did the last fetch. That is, if a store succeeds then no
53 // other writes have been performed on the variable since your fetch.
54 //
55 // Example:
56 //
57 // Storage* storage = new ZooKeeperStorage();
58 // State* state = new State(storage);
59 // Future<Variable> variable = state->fetch("slaves");
60 // std::string value = update(variable.value());
61 // variable = variable.mutate(value);
62 // state->store(variable);
63

157 inline process::Future<Option<Variable> > State::store(const Variable& variable)
158 {
159  // Note that we try and swap an entry even if the value didn't change!
160  UUID uuid = UUID::fromBytes(variable.entry.uuid());
161 
162  // Create a new entry to replace the existing entry provided the
163  // UUID matches.
164  Entry entry;
165  entry.set_name(variable.entry.name());
166  entry.set_uuid(UUID::random().toBytes());
167  entry.set_value(variable.entry.value());
168 
169  return storage->set(entry, uuid)
170  .then(lambda::bind(&State::_store, entry, lambda::_1));
171 }
