38 // change most of the callsites that use `UUID` to use `id::UUID` instead.

55  static Try<UUID> fromBytes(const std::string& s)
56  {
57  const std::string error = "Not a valid UUID";
58 
59  if (s.size() != UUID::static_size()) {
60  return Error(error);
61  }
62 
63  boost::uuids::uuid uuid;
64  memcpy(&uuid, s.data(), s.size());
65 
66  if (uuid.version() == UUID::version_unknown) {
67  return Error(error);
68  }
69 
70  return UUID(uuid);
71  }

73  static Try<UUID> fromString(const std::string& s)
74  {
75  try {
76  // NOTE: We don't use THREAD_LOCAL for the `string_generator`
77  // (unlike for the `random_generator` above), because it is cheap
78  // to construct one each time.
79  boost::uuids::string_generator gen;
80  boost::uuids::uuid uuid = gen(s);
81  return UUID(uuid);
82  } catch (const std::runtime_error& e) {
83  return Error(e.what());
84  }
85  }
