26 // When a debugger is attached to the process, `::CloseHandle` can throw a
27 // structured exception under some circumstances (for example, when we pass it
28 // an invalid handle to close). This function will catch the exception so that
29 // our test runs don't get interrupted when we happen to run them with a
30 // debugger.
31 //
32 // NOTE: Structured exceptions emitted by the C runtime APIs can't be caught
33 // using C++ exceptions by default, and enabling this requires adding special
34 // build flags. Since we don't gain anything by using this feature, we choose
35 // to just use the default behavior. Note also that structured exceptions must
36 // be caught in functions that do not require object unwinding (e.g., no
37 // destructors need to be called), so we make it a standalone, isolated
38 // function.
39 inline bool safe_closehandle(HANDLE handle)
40 {
41  BOOL closed = false;
42  __try
43  {
44  closed = ::CloseHandle(handle);
45  }
46  __except (EXCEPTION_EXECUTE_HANDLER)
47  {
48  ::SetLastError(ERROR_INVALID_HANDLE);
49  }
50 
51  return static_cast<bool>(closed);
52 }

55 // A version of the POSIX-standard function, `close`, that catches structured
56 // exceptions thrown by the CRT. In these cases, we instead return `-1` and set
57 // `errno` to `EBADF`.
58 //
59 // `safe_close` is intended as an important, last-ditch attempt to avoid
60 // throwing a process-halting exception if we accidentally call `_close` on a
61 // file descriptor that corresponds to a socket. There are checks meant to
62 // prevent this in the `os::close` family as well; this is just our last
63 // defense.
64 //
65 // NOTE: Structured exceptions emitted by the C runtime APIs can't be caught
66 // using C++ exceptions by default, and enabling this requires adding special
67 // build flags. Since we don't gain anything by using this feature, we choose
68 // to just use the default behavior. Note also that structured exceptions must
69 // be caught in functions that do not require object unwinding (e.g., no
70 // destructors need to be called), so we make it a standalone, isolated
71 // function.
72 inline int safe_close(int fd)
73 {
74  int close = -1;
75  __try
76  {
77  close = ::_close(fd);
78  }
79  __except (EXCEPTION_EXECUTE_HANDLER)
80  {
81  ::_set_errno(EBADF);
82  }
83 
84  return close;
85 }

103 inline Try<Nothing> close(int fd)
104 {
105  // NOTE: This is a defensive measure. Since `int` is used throughout the
106  // codebase to represent file descriptors, including those which refer to
107  // sockets, it is possible that we could accidentally cast a `SOCKET` to an
108  // `int` and pass it in here. Thus, we check explicitly whether `fd` is a
109  // socket, and dispatch to the appropriate `close` function.
110  //
111  // The most notable example of this problem is `socket.hpp`, which represents
112  // the socket using an `int`. Changing this would require major surgery
113  // throughout the codebase.
114  //
115  // Note also that casting a file descriptor to a `SOCKET`, however, is less
116  // likely. Hence it is only necessary to cast in this variant of the function.
117  //
118  // TODO(hausdorff): Transition away from this defensive measure when we
119  // transition from `int` to an encapsulated file descriptor type.
120  if (net::is_socket(static_cast<SOCKET>(fd))) {
121  return close(static_cast<SOCKET>(fd));
122  } else {
123  if (::_close(fd) == -1) {
124  return ErrnoError("os::close: Failed to close file descriptor");
125  }
126  }
127 
128  return Nothing();
129 }
