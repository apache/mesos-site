23 // Not using extern as this is used only by the executable. The signal handler

28 BOOL WINAPI CtrlHandler(DWORD fdwCtrlType)
29 {
30  switch (fdwCtrlType) {
31  // Handle the CTRL-C signal.
32  case CTRL_C_EVENT:
33  case CTRL_CLOSE_EVENT:
34  case CTRL_BREAK_EVENT:
35  case CTRL_LOGOFF_EVENT:
36  case CTRL_SHUTDOWN_EVENT: {
37  if (signaledWrapper != nullptr) {
38  (*signaledWrapper)(SIGUSR1, 0);
39  return TRUE;
40  }
41  }
42  default: {
43  return FALSE;
44  }
45  }
46 }
