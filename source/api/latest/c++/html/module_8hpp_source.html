62 template <typename T>
63 bool readPythonProtobuf(PyObject* obj, T* t)
64 {
65  if (obj == Py_None) {
66  std::cerr << "None object given where protobuf expected" << std::endl;
67  return false;
68  }
69  PyObject* res = PyObject_CallMethod(obj,
70  (char*) "SerializeToString",
71  (char*) NULL);
72  if (res == NULL) {
73  std::cerr << "Failed to call Python object's SerializeToString "
74  << "(perhaps it is not a protobuf?)" << std::endl;
75  PyErr_Print();
76  return false;
77  }
78  char* chars;
79  Py_ssize_t len;
80  if (PyString_AsStringAndSize(res, &chars, &len) < 0) {
81  std::cerr << "SerializeToString did not return a string" << std::endl;
82  PyErr_Print();
83  Py_DECREF(res);
84  return false;
85  }
86  google::protobuf::io::ArrayInputStream stream(chars, len);
87  bool success = t->ParseFromZeroCopyStream(&stream);
88  if (!success) {
89  std::cerr << "Could not deserialize protobuf as expected type" << std::endl;
90  }
91  Py_DECREF(res);
92  return success;
93 }

104 {
105  PyObject* dict = PyModule_GetDict(mesos_pb2);
106  if (dict == NULL) {
107  PyErr_Format(PyExc_Exception, "PyModule_GetDict failed");
108  return NULL;
109  }
110 
111  PyObject* type = PyDict_GetItemString(dict, typeName);
112  if (type == NULL) {
113  PyErr_Format(PyExc_Exception, "Could not resolve mesos_pb2.%s", typeName);
114  return NULL;
115  }
116  if (!PyType_Check(type)) {
117  PyErr_Format(PyExc_Exception, "mesos_pb2.%s is not a type", typeName);
118  return NULL;
