88  {
89  CHECK(sizeof(value) == 8);
90  char bytes[8];
91  bytes[0] =(0xff & (value >> 56));
92  bytes[1] = (0xff & (value >> 48));
93  bytes[2] = (0xff & (value >> 40));

95  bytes[4] = (0xff & (value >> 24));
96  bytes[5] = (0xff & (value >> 16));
97  bytes[6] = (0xff & (value >> 8));
98  bytes[7] = (0xff & value);
