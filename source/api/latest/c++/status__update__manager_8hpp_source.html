137 // StatusUpdateStream handles the status updates and acknowledgements
138 // of a task, checkpointing them if necessary. It also holds the information
139 // about received, acknowledged and pending status updates.
140 // NOTE: A task is expected to have a globally unique ID across the lifetime
141 // of a framework. In other words the tuple (taskId, frameworkId) should be
142 // always unique.
143 struct StatusUpdateStream
144 {
145  StatusUpdateStream(const TaskID& _taskId,
146  const FrameworkID& _frameworkId,
147  const SlaveID& _slaveId,
148  const Flags& _flags,
149  bool _checkpoint,
150  const Option<ExecutorID>& executorId,
151  const Option<ContainerID>& containerId)
152  : checkpoint(_checkpoint),
153  terminated(false),
154  taskId(_taskId),
155  frameworkId(_frameworkId),
156  slaveId(_slaveId),
157  flags(_flags),
158  error(None())
159  {
160  if (checkpoint) {
161  CHECK_SOME(executorId);
162  CHECK_SOME(containerId);
163 
164  path = paths::getTaskUpdatesPath(
165  paths::getMetaRootDir(flags.work_dir),
166  slaveId,
167  frameworkId,
168  executorId.get(),
169  containerId.get(),
170  taskId);
171 
172  // Create the base updates directory, if it doesn't exist.
173  Try<Nothing> directory = os::mkdir(os::dirname(path.get()).get());
174  if (directory.isError()) {
175  error = "Failed to create " + os::dirname(path.get()).get();
176  return;
177  }
178 
179  // Open the updates file.
180  Try<int> result = os::open(
181  path.get(),
182  O_CREAT | O_WRONLY | O_APPEND | O_SYNC,
183  S_IRUSR | S_IWUSR | S_IRGRP | S_IRWXO);
184 
185  if (result.isError()) {
186  error = "Failed to open '" + path.get() + "' for status updates";
187  return;
188  }
189 
190  // We keep the file open through the lifetime of the task, because it
191  // makes it easy to append status update records to the file.
192  fd = result.get();
193  }
194  }

196  ~StatusUpdateStream()

198  if (fd.isSome()) {
199  Try<Nothing> close = os::close(fd.get());

208  // This function handles the update, checkpointing if necessary.
209  // @return True if the update is successfully handled.
210  // False if the update is a duplicate.
211  // Error Any errors (e.g., checkpointing).
212  Try<bool> update(const StatusUpdate& update)
213  {
214  if (error.isSome()) {
215  return Error(error.get());
216  }
217 
218  // Check that this status update has not already been acknowledged.
219  // This could happen in the rare case when the slave received the ACK
220  // from the framework, died, but slave's ACK to the executor never made it!
221  if (acknowledged.contains(UUID::fromBytes(update.uuid()))) {
222  LOG(WARNING) << "Ignoring status update " << update
223  << " that has already been acknowledged by the framework!";
224  return false;
225  }
226 
227  // Check that this update hasn't already been received.
228  // This could happen if the slave receives a status update from an executor,
229  // then crashes after it writes it to disk but before it sends an ack.
230  if (received.contains(UUID::fromBytes(update.uuid()))) {
231  LOG(WARNING) << "Ignoring duplicate status update " << update;
232  return false;
233  }
234 
235  // Handle the update, checkpointing if necessary.
236  Try<Nothing> result = handle(update, StatusUpdateRecord::UPDATE);
237  if (result.isError()) {
238  return Error(result.error());
239  }
240 
241  return true;
242  }

244  // This function handles the ACK, checkpointing if necessary.
245  // @return True if the acknowledgement is successfully handled.
246  // False if the acknowledgement is a duplicate.
247  // Error Any errors (e.g., checkpointing).
248  Try<bool> acknowledgement(
249  const TaskID& taskId,
250  const FrameworkID& frameworkId,
251  const UUID& uuid,
252  const StatusUpdate& update)
253  {
254  if (error.isSome()) {
255  return Error(error.get());
256  }
257 
258  if (acknowledged.contains(uuid)) {
259  LOG(WARNING) << "Duplicate status update acknowledgment (UUID: "
260  << uuid << ") for update " << update;
261  return false;
262  }
263 
264  // This might happen if we retried a status update and got back
265  // acknowledgments for both the original and the retried update.
266  if (uuid != UUID::fromBytes(update.uuid())) {
267  LOG(WARNING) << "Unexpected status update acknowledgement (received "
268  << uuid << ", expecting " << UUID::fromBytes(update.uuid())
269  << ") for update " << update;
270  return false;
271  }
272 
273  // Handle the ACK, checkpointing if necessary.
274  Try<Nothing> result = handle(update, StatusUpdateRecord::ACK);
275  if (result.isError()) {
276  return Error(result.error());
277  }
278 
279  return true;
280  }

296  // Replays the stream by sequentially handling an update and its
297  // corresponding ACK, if present.
298  Try<Nothing> replay(
299  const std::vector<StatusUpdate>& updates,
300  const hashset<UUID>& acks)
301  {
302  if (error.isSome()) {
303  return Error(error.get());
304  }
305 
306  VLOG(1) << "Replaying status update stream for task " << taskId;
307 
308  foreach (const StatusUpdate& update, updates) {
309  // Handle the update.
310  _handle(update, StatusUpdateRecord::UPDATE);
311 
312  // Check if the update has an ACK too.
313  if (acks.contains(UUID::fromBytes(update.uuid()))) {
314  _handle(update, StatusUpdateRecord::ACK);
315  }
316  }
317 
318  return Nothing();
319  }

328  // Handles the status update and writes it to disk, if necessary.
329  // TODO(vinod): The write has to be asynchronous to avoid status updates that
330  // are being checkpointed, blocking the processing of other updates.
331  // One solution is to wrap the protobuf::write inside async, but its probably
332  // too much of an overhead to spin up a new libprocess per status update?
333  // A better solution might be to be have async write capability for file io.
334  Try<Nothing> handle(
335  const StatusUpdate& update,
336  const StatusUpdateRecord::Type& type)
337  {
338  CHECK(error.isNone());
339 
340  // Checkpoint the update if necessary.
341  if (checkpoint) {
342  LOG(INFO) << "Checkpointing " << type << " for status update " << update;
343 
344  CHECK_SOME(fd);
345 
346  StatusUpdateRecord record;
347  record.set_type(type);
348 
349  if (type == StatusUpdateRecord::UPDATE) {
350  record.mutable_update()->CopyFrom(update);
351  } else {
352  record.set_uuid(update.uuid());
353  }
354 
355  Try<Nothing> write = ::protobuf::write(fd.get(), record);
356  if (write.isError()) {
357  error = "Failed to write status update " + stringify(update) +
358  " to '" + path.get() + "': " + write.error();
359  return Error(error.get());
360  }
361  }
362 
363  // Now actually handle the update.
364  _handle(update, type);
365 
366  return Nothing();
367  }

369  void _handle(const StatusUpdate& update, const StatusUpdateRecord::Type& type)
370  {
371  CHECK(error.isNone());
372 
373  if (type == StatusUpdateRecord::UPDATE) {
374  // Record this update.
375  received.insert(UUID::fromBytes(update.uuid()));
376 
377  // Add it to the pending updates queue.
378  pending.push(update);
379  } else {
380  // Record this ACK.
381  acknowledged.insert(UUID::fromBytes(update.uuid()));
382 
383  // Remove the corresponding update from the pending queue.
384  pending.pop();
385 
386  if (!terminated) {
387  terminated = protobuf::isTerminalState(update.status().state());
388  }
389  }
390  }
